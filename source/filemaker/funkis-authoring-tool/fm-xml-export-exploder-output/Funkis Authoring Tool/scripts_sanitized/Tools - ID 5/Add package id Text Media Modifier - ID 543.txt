# This script set the package id reference for all text and media the package where you stand package. It uses the parent id to find texts 
Set Error Capture [ ON ]

Set Variable [ $title_id[1] ; Case( GetScriptProperty ( "title_id" )= ""; FAT Title::PrimaryKey; GetScriptProperty ( "title_id" )) ]
Set Variable [ $package_id[1] ; Case( GetScriptProperty ( "package_id" )= ""; FAT Package::PrimaryKey; GetScriptProperty ( "package_id" )) ]

Go to Layout [ Layout: "TABLE FAT Title" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

# Find all modifierValues having text or media
# Package
# Package Modifier media
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Go to Related Record [ Table: Media MV PK ; Layout: "TABLE FAT Media" ; Show related ; Match found set ]
If [ Get(foundCount)>0 ]
	Replace Field Contents [ FAT Media::Package id ; Replace with calculation: $package_id ]
End If

# Package Modifier text
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Go to Related Record [ Table: TF MV PK ; Layout: "TABLE FAT Text" ; Show related ; Match found set ]
If [ Get(foundCount)>0 ]
	Replace Field Contents [ FAT Text::Package id ; Replace with calculation: $package_id ]
End If

# Chapter
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Go to Related Record [ Table: FAT Chapter ; Layout: "TABLE FAT Chapter" ; Show related ; Match found set ]
If [ Get(foundCount)>0 ]
	Loop [ Flush: Always ]
		# Chapter Modifier media
		Go to Related Record [ Table: Media MV CH ; Layout: "TABLE FAT Media" ; Show related ; Match found set ]
		If [ Get(foundCount)>0 ]
			Replace Field Contents [ FAT Media::Package id ; Replace with calculation: $package_id ]
		End If
		# Chapter Modifier text
		Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
		Go to Related Record [ Table: TF MV CH ; Layout: "TABLE FAT Text" ; Show related ; Match found set ]
		If [ Get(foundCount)>0 ]
			Replace Field Contents [ FAT Text::Package id ; Replace with calculation: $package_id ]
		End If
		# Pages
		Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
		Go to Related Record [ Table: FAT Page ; Layout: "TABLE FAT Page" ; Show related ; Match found set ]
		If [ Get(foundCount)>0 ]
			Loop [ Flush: Always ]
				# Page Modifier media
				Go to Related Record [ Table: Media MV PG ; Layout: "TABLE FAT Media" ; Show related ; Match found set ]
				If [ Get(foundCount)>0 ]
					Replace Field Contents [ FAT Media::Package id ; Replace with calculation: $package_id ]
				End If
				# Page Modifier text
				Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
				Go to Related Record [ Table: TF MV PG ; Layout: "TABLE FAT Text" ; Show related ; Match found set ]
				If [ Get(foundCount)>0 ]
					Replace Field Contents [ FAT Text::Package id ; Replace with calculation: $package_id ]
				End If
				
				# Page items
				Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
				Go to Related Record [ Table: FAT Page item ; Layout: "TABLE FAT Page Item" ; Show related ; Match found set ]
				If [ Get(foundCount)>0 ]
					Loop [ Flush: Always ]
						# Page item attribute media
						Go to Related Record [ Table: Media MV PI ; Layout: "TABLE FAT Media" ; Show related ; Match found set ]
						If [ Get(foundCount)>0 ]
							Replace Field Contents [ FAT Media::Package id ; Replace with calculation: $package_id ]
						End If
						Go to Layout [ Layout: "TABLE FAT Page Item" ; Animation: None ]
						# Page item attribute text
						Go to Related Record [ Table: TF MV PI ; Layout: "TABLE FAT Text" ; Show related ; Match found set ]
						If [ Get(foundCount)>0 ]
							Replace Field Contents [ FAT Text::Package id ; Replace with calculation: $package_id ]
						End If
						
						Go to Layout [ Layout: "TABLE FAT Page Item" ; Animation: None ]
						Go to Record/Request/Page [ Next ; Exit after last: OFF ]
					End Loop
				End If
				
				Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
				Go to Record/Request/Page [ Next ; Exit after last: OFF ]
			End Loop
		End If
		
		# When all media package id are set - find all media mp4 having subtitles in package
		Go to Layout [ Layout: "TABLE FAT Media" ; Animation: None ]
		Perform Find [ ⚠️ RESTORE ⚠️ ]
		If [ Get(foundCount)>0 ]
			Loop [ Flush: Always ]
				Set Variable [ $item_id[1] ; FAT Media::PrimaryKey ]
				Go to Layout [ Layout: "TABLE FAT Text" ; Animation: None ]
				Perform Find [ ⚠️ RESTORE ⚠️ ]
				Replace Field Contents [ FAT Text::Package id ; Replace with calculation: $package_id ]
				Go to Layout [ Layout: "TABLE FAT Media" ; Animation: None ]
				Go to Record/Request/Page [ Next ; Exit after last: ON ]
			End Loop
		End If
	End Loop
End If
Go to Layout [ Layout: <original layout> ; Animation: None ]
