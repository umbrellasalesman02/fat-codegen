Set Error Capture [ ON ]
Go to Layout [ Layout: "TABLE FAT Title" ; Animation: None ]
Set Variable [ $lpid[1] ; FAT Title::PrimaryKey ]
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
Loop [ Flush: Always ]
	Set Variable [ $course_id[1] ; FAT Package::PrimaryKey ]
	Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $masterchapter_id[1] ; FAT Chapter::PrimaryKey ]
	Set Field [ FAT Chapter::Proxy type ; "" ]
	Set Field [ FAT Chapter::Hierarchy level ; 2 ]
	Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Field [ FAT Page::Hierarchy level ; 3 ]
	Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
	If [ Get(foundCount)>1 ]
		Go to Record/Request/Page [ Next ; Exit after last: OFF ]
		Loop [ Flush: Always ]
			Set Variable [ $chapter_id[1] ; FAT Chapter::PrimaryKey ]
			Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			If [ Get(foundCount)>0 ]
				Set Field [ FAT Page::Parent id ; $masterchapter_id ]
				Set Field [ FAT Page::Hierarchy level ; 3 ]
			End If
			Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
			Set Field [ FAT Chapter::Proxy type ; "DELETE" ]
			Set Field [ FAT Chapter::Hierarchy level ; 2 ]
			Go to Record/Request/Page [ Next ; Exit after last: ON ]
		End Loop
	End If
	Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
	Go to Record/Request/Page [ Next ; Exit after last: ON ]
End Loop
