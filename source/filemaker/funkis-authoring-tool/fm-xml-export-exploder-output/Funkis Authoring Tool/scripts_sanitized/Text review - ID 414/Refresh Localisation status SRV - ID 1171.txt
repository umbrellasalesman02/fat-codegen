
# This script is used in review by trasnlators when changing text status.
  It is called with Wait for completetion = OFF

# Text status can be:
  1. Not translated
  2. Sent to API (only asynchronous machine translation)
  3. Not reviewed
  4. In progress
  5. Approved

Set Variable [ $mode[1] ; GetScriptProperty ( "mode" ) ]
Set Variable [ $title_id[1] ; GetScriptProperty ( "title_id" ) ]
Set Variable [ $package_id[1] ; GetScriptProperty ( "package_id" ) ]
Set Variable [ $localisation_code[1] ; GetScriptProperty ( "localisation_code" ) ]
Set Variable [ $shared_localisation[1] ; GetAsBoolean(GetScriptProperty ( "shared_localisation" )) ]

Freeze Window

Set Variable [ $status_list[1] ; "Not translated¶Sent to API¶Not reviewed¶In progress¶Approved" ]
If [ $mode = "Title" ]
	# Update fields in Title
	Go to Layout [ Layout: "TABLE FAT Title" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $field_list[1] ; GetFieldName ( FAT Title::TFs Not_translated ) &"¶"&
					GetFieldName ( FAT Title::TFs Sent_to_api )&"¶"&
					GetFieldName ( FAT Title::TFs Not_reviewed )&"¶"&
					GetFieldName ( FAT Title::TFs In_progress )&"¶"&
					GetFieldName ( FAT Title::TFs Approved ) ]
Else
	# Find the right localisation to update
	Go to Layout [ Layout: "TABLE FAT Localisation" ; Animation: None ]
	If [ IsEmpty ( $package_id ) And Not $shared_localisation ]
		# Title localisation
		Perform Find [ ⚠️ RESTORE ⚠️ ]
	Else If [ IsEmpty ( $package_id ) And $shared_localisation ]
		# Shared localisation
		Perform Find [ ⚠️ RESTORE ⚠️ ]
	Else
		# Package localisation
		Perform Find [ ⚠️ RESTORE ⚠️ ]
	End If
	Set Variable [ $field_list[1] ; GetFieldName ( FAT Localisation::TFs Not_translated ) &"¶"&
					GetFieldName ( FAT Localisation::TFs Sent_to_api )&"¶"&
					GetFieldName ( FAT Localisation::TFs Not_reviewed )&"¶"&
					GetFieldName ( FAT Localisation::TFs In_progress )&"¶"&
					GetFieldName ( FAT Localisation::TFs Approved ) ]
End If

Set Variable [ $i[1] ; 1 ]

Loop [ Flush: Always ]
	Set Variable [ $status[1] ; GetValue ( $status_list ; $i ) ]
	Set Variable [ $num_of_texts[1] ; Let ( [
					
					    //input
					    lc=$localisation_code;
					    title = $title_id;
					    status = $status;
					    package = $package_id;
					    shared_localisation = $shared_localisation;
					
					    //count field
					    cf = #F ( FAT Text::PrimaryKey );
					    
					
					    //from table
					    table = #T ( FAT Text::PrimaryKey);
					
					    //where fields
					
					    lc_f = #F ( FAT Text::Localisation code );
					    status_f = #F ( FAT Text::Status );
					    title_f = #F ( FAT Text::Title id );
					    package_f = #F ( FAT Text::Package id );
					
					    ~query = Case(
					
					       package = "" And Not shared_localisation; 
					
					          "SELECT COUNT(~cf) FROM ~table WHERE ~lc_f=? AND ~title_f=? AND ~status_f=?";
					
					       package = "" And shared_localisation; 
					
					          "SELECT COUNT(~cf) FROM ~table WHERE ~lc_f=? AND ~title_f=? AND ~status_f=? AND ~package_f IS NULL";
					
					
					          "SELECT COUNT(~cf) FROM ~table WHERE ~lc_f=? AND ~title_f=? AND ~status_f=? AND ~package_f=?"
					    );
					
					~sqlQuery = Substitute (  ~query ; 
					            ["~cf"; cf];
					            ["~table"; table];
					            ["~lc_f"; lc_f];
					            ["~title_f"; title_f];
					            ["~package_f"; package_f];
					            ["~status_f"; status_f]
					 )
					
					   
					
					];
					
					Case(package = ""; 
					
					    ExecuteSQL (~sqlQuery ; "" ; "" ; lc; title; status );
					
					    ExecuteSQL (~sqlQuery ; "" ; "" ; lc; title; status; package )
					)
					
					//~sqlQuery & ", " & lc & ", " & title & ", " & status
					
					 ) ]
	
	Set Variable [ $field_name[1] ; GetValue($field_list; $i) ]
	Set Field By Name [ Specify target field: ON ; $field_name ; $num_of_texts ]
	Set Variable [ $i[1] ; $i + 1 ]
	Exit Loop If [ $i > 5 ]
End Loop
Commit Records/Requests [ With dialog: OFF ]
Go to Layout [ Layout: <original layout> ; Animation: None ]
