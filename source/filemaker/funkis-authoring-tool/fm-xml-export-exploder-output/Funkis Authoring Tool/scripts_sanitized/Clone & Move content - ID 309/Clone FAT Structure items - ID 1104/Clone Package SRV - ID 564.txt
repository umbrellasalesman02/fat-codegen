# This script clones a specific Package and all of its Chapters, Pages, Page items, Modifiers, Tags, Texts, Media etc
# This script is called from:
  1. Clone Title
  2. Clone Package group
  3. From a button in Package Settings (via script Clone Package CL)
  4. From script Start Cloning
# Script is designed to work PSOS

# SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
  SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
  SetProperty ( "clone_level" ; $clone_level ) &
  SetProperty ( "clone_settings" ; $clone_settings ) &
  SetProperty ( "suffix" ; $suffix) &
  SetProperty ( "title_id" ; $title_id ) &
  SetProperty ( "package_group_id" ; $new_pkgr_id ) &
  SetProperty ( "object_group_id" ; $object_group_id ) &
  SetProperty ( "package_id" ; $package_id ) &
  SetProperty ( "new_sortindex" ; $new_sortindex )


Freeze Window
Set Error Capture [ ON ]

# If the whole Title is cloned we need to point to new Blocks, Modifiers, Lists, Links, Score Categories, IOs, Tags etc
Set Variable [ $isClonedTitle[1] ; GetAsBoolean(GetScriptProperty ( "isClonedTitle")) ]
Set Variable [ $isMovedPackage[1] ; GetAsBoolean(GetScriptProperty ( "isMovedPackage" )) ]

Set Variable [ $clone_level[1] ; GetScriptProperty ( "clone_level" ) ]
Set Variable [ $clone_settings[1] ; GetScriptProperty ( "clone_settings" ) ]
Set Variable [ $suffix[1] ; GetScriptProperty ( "suffix" ) ]
Set Variable [ $object_group_id[1] ; GetScriptProperty ( "object_group_id" ) ]

Set Variable [ $title_id[1] ; GetScriptProperty ( "title_id") ]
Set Variable [ $package_group_id[1] ; GetScriptProperty ( "package_group_id") ]

Set Variable [ $old_pk_id[1] ; GetScriptProperty ( "package_id") ]
Set Variable [ $new_sortindex[1] ; GetAsNumber(GetScriptProperty ( "new_sortindex")) ]

# Find the package
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

# Save current data
Set Variable [ $pk_title_text[1] ; Case(Not IsEmpty($suffix); TF PK Header::Text & " " & $suffix; TF PK Header::Text ) ]
Set Variable [ $pk_subtitle1_text[1] ; TF PK Sub header 1::Text ]
Set Variable [ $pk_subtitle2_text[1] ; TF PK Sub header 2::Text ]
Set Variable [ $pk_sortindex[1] ; Case ($new_sortindex; $new_sortindex ; FAT Package::SortIndex) ]
Set Variable [ $pk_reference_name[1] ; Case(Not IsEmpty($suffix); FAT Package::Reference name  & " " & $suffix; FAT Package::Reference name) ]
Set Variable [ $pk_fc_package_directory[1] ; Case ( FAT Package::FC Package directory = "title-information"; 
				          "title-information" ; 
				           PrettifyDash ( FAT Package::FC Package directory & "-" & FAT Package::Serial ID)
				      ) ]
Set Variable [ $package_type[1] ; FAT Package::Package type id ]

# Define value of variable $package_group_id
If [ $isClonedTitle And FAT Package::Package Group ≠ "" ]
	Set Variable [ $old_pkgr_id[1] ; FAT Package::Package Group ]
	Go to Layout [ Layout: "TABLE FAT Package Group" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $package_group_id[1] ; FAT Package Group::PrimaryKey ]
	Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Else If [ $isMovedPackage And Not IsEmpty ( $package_group_id ) ]
	# Moving a package when moving a package group - do nothing - $package_group_id is the NEW package_group_id
Else If [ $isMovedPackage And IsEmpty ( $package_group_id ) ]
	# Moving a single package
	# Moved single packages should not have a package group even if they had it initially. 
	Set Variable [ $package_group_id[1] ; "" ]
Else If [ $clone_level = "Package group" ]
	# $package_group_id set from Clone Package Group
Else
	Set Variable [ $package_group_id[1] ; FAT Package::Package Group ]
End If

Duplicate Record/Request

Set Variable [ $new_pk_id[1] ; FAT Package::PrimaryKey ]
# Update cloned record with new data
Set Field [ FAT Package::Old PrimaryKey from clone ; $old_pk_id ]
Set Field [ FAT Package::Parent id ; $title_id ]
Set Field [ FAT Package::SortIndex ; $pk_sortindex ]
Set Field [ FAT Package::Reference name ; $pk_reference_name ]
Set Field [ FAT Package::FC Package directory ; $pk_fc_package_directory ]
Set Field [ FAT Package::Package Group ; $package_group_id ]

Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; $new_pk_id ) &
				SetProperty ( "source_table" ; "package" ) &
				SetProperty ( "source_textfield" ; GetFieldName ( FAT Package::Header txt id ) ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $new_pk_id ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]
Set Field [ TF PK Header::Text ; $pk_title_text ]
Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; $new_pk_id ) &
				SetProperty ( "source_table" ; "package" ) &
				SetProperty ( "source_textfield" ; GetFieldName ( FAT Package::Sub header 1 txt id ) ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $new_pk_id ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]
Set Field [ TF PK Sub header 1::Text ; $pk_subtitle1_text ]
Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; $new_pk_id ) &
				SetProperty ( "source_table" ; "package" ) &
				SetProperty ( "source_textfield" ; GetFieldName ( FAT Package::Sub header 2 txt id ) ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $new_pk_id ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]
Set Field [ TF PK Sub header 2::Text ; $pk_subtitle2_text ]

# Add to content overview - NEW 2024-02-10
Set Variable [ $item_type[1] ; "Package" ]
Set Variable [ $item_id[1] ; $new_pk_id ]
Perform Script [ From list ; "Add Overview item SRV" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "item_id" ; $item_id ) &
				SetProperty ( "item_type" ; $item_type ) ]

# Check if Link to this item exist. Duplicate Link if $clone_settings include Link
If [ PatternCount ( $clone_settings ; "Link" ) ]
	Perform Script [ From list ; "Clone Link SRV" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
					SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
					SetProperty ( "clone_level" ; $clone_level ) &
					SetProperty ( "clone_settings" ; $clone_settings ) &
					SetProperty ( "object_group_id" ; $object_group_id ) &
					SetProperty ( "title_id" ; $title_id ) &
					SetProperty ( "suffix" ; $suffix) &
					SetProperty ( "old_item_id" ; $old_pk_id ) &
					SetProperty ( "new_item_id" ; $new_pk_id ) ]
End If

Perform Script [ From list ; "Clone ModifierValue" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
				SetProperty ( "isMovedPackage" ; $isMovedPackage ) & 
				SetProperty ( "old_item_id" ; $old_pk_id ) & 
				SetProperty ( "new_item_id" ; $new_pk_id ) & 
				SetProperty ( "title_id" ; $title_id ) & 
				SetProperty ( "package_id" ; $new_pk_id ) ]
Perform Script [ From list ; "Clone Tag values" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
				SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
				SetProperty ( "clone_level" ; $clone_level ) &
				SetProperty ( "clone_settings" ; $clone_settings ) &
				SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "suffix" ; $suffix) &
				SetProperty ( "object_group_id" ; $object_group_id ) &
				SetProperty ( "old_item_id" ; $old_pk_id ) & 
				SetProperty ( "new_item_id" ; $new_pk_id ) ]
Perform Script [ From list ; "Clone Score values" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
				SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
				SetProperty ( "clone_level" ; $clone_level ) &
				SetProperty ( "clone_settings" ; $clone_settings ) &
				SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "suffix" ; $suffix) &
				SetProperty ( "object_group_id" ; $object_group_id ) &
				SetProperty ( "old_item_id" ; $old_pk_id ) & 
				SetProperty ( "new_item_id" ; $new_pk_id ) ]


# Go to Chapters
# Callers always usde the Alt TO
Go to Layout [ Layout: "TABLE FAT Chapter Alt" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	Sort Records [ With dialog: OFF ; ⚠️ PARAMETER "Restore" NOT PARSED ⚠️ ; ⚠️ PARAMETER "SortSpecification" NOT PARSED ⚠️ ]
	Go to Record/Request/Page [ First ]
	Loop [ Flush: Always ]
		# Chapter loop
		Set Variable [ $old_ch_id[1] ; FAT Chapter Alt::PrimaryKey ]
		Set Variable [ $old_sortindex[1] ; FAT Chapter Alt::SortIndex ]
		Perform Script [ From list ; "Clone Chapter SRV" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
						SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
						SetProperty ( "clone_level" ; $clone_level) &
						SetProperty ( "clone_settings" ; $clone_settings ) &
						SetProperty ( "suffix" ; $suffix) &
						SetProperty ( "object_group_id" ; $object_group_id ) &
						SetProperty ( "title_id" ; $title_id ) &
						SetProperty ( "parent_id" ; $new_pk_id ) &
						SetProperty ( "old_ch_id" ; $old_ch_id ) ]
		Go to Layout [ Layout: "TABLE FAT Chapter Alt" ; Animation: None ]
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
End If

Exit Script [ $item_id ]
