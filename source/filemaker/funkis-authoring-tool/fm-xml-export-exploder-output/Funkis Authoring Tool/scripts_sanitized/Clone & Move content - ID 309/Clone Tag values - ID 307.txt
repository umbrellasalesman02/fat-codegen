# This script is called by Clone content SRV  for each item that can have a tag


Set Variable [ $isClonedTitle[1] ; GetAsBoolean(GetScriptProperty ( "isClonedTitle")) ]
Set Variable [ $isMovedPackage[1] ; GetAsBoolean(GetScriptProperty ( "isMovedPackage" )) ]

Set Variable [ $clone_level[1] ; GetScriptProperty ( "clone_level") ]
Set Variable [ $clone_settings[1] ; GetScriptProperty ( "clone_settings" ) ]
Set Variable [ $suffix[1] ; GetScriptProperty ( "suffix" ) ]
Set Variable [ $object_group_id[1] ; GetScriptProperty ( "object_group_id" ) ]

Set Variable [ $title_id[1] ; GetScriptProperty ( "title_id") ]
Set Variable [ $old_item_id[1] ; GetScriptProperty ( "old_item_id") ]
Set Variable [ $new_item_id[1] ; GetScriptProperty ( "new_item_id") ]

# -----------------------------------------------------------------------------
# --------------- TAG --------------------------------------------------------

# The process is the following:
  1. Duplicate the tag value
  2. If clone settings includes Tags do the following: 
  3. Check if new Tag is already created by checking OldPrimaryKeyFromClone. If found - connect this Tag id
  4. If not found - call script "Clone Tag" which will return a new Tag id
  5. Clone Tag will duplicate full Tag Filter list, including List items and Tags


Set Error Capture [ ON ]
Go to Layout [ Layout: "TABLE FAT Tag Value" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	Set Variable [ $foundCount[1] ; Get(foundCount) ]
	Set Variable [ $recordPosn[1] ; 1 ]
	Go to Record/Request/Page [ First ]
	Loop [ Flush: Always ]
		Duplicate Record/Request
		Set Variable [ $recordPosn[1] ; $recordPosn  + 1 ]
		Set Field [ FAT Tag Value::Parent id ; $new_item_id ]
		Set Variable [ $tag_value_id[1] ; FAT Tag Value::PrimaryKey ]
		Set Variable [ $old_tag_id[1] ; FAT Tag Value::Tag id ]
		
		# Do not duplicate funkis cloud tag lists.
		If [ $isClonedTitle Or (PatternCount($clone_settings; "Tags") And Not PatternCount ( FAT Tag::List id for JSON ; "funkis") ) ]
			Go to Layout [ Layout: "TABLE FAT Tag" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			If [ Get(foundCount) = 1 ]
				Set Variable [ $new_tag_id[1] ; FAT Tag::PrimaryKey ]
				Go to Layout [ Layout: "TABLE FAT Tag Value" ; Animation: None ]
				Set Field [ FAT Tag Value::Tag id ; $new_tag_id ]
			Else If [ Get(foundCount) = 0 ]
				# No oldPrimaryKey found. Create a new Tag and add it to the tag value
				Perform Script [ From list ; "Clone Tag SRV" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
								SetProperty ( "clone_level" ; $clone_level ) &
								SetProperty ( "clone_settings" ; $clone_settings ) &
								SetProperty ( "suffix" ; $suffix ) &
								SetProperty ( "object_group_id" ; $object_group_id ) &
								SetProperty ( "title_id" ; $title_id ) &
								SetProperty ( "old_tag_id" ; $old_tag_id ) ]
				Set Variable [ $new_tag_id[1] ; Get(ScriptResult) ]
				Go to Layout [ Layout: "TABLE FAT Tag Value" ; Animation: None ]
				Set Field [ FAT Tag Value::Tag id ; $new_tag_id ]
			End If
		End If
		
		Go to Layout [ Layout: "TABLE FAT Tag Value" ; Animation: None ]
		Exit Loop If [ $recordPosn > $foundCount ]
		Go to Record/Request/Page [ With dialog: OFF ; $recordPosn ]
	End Loop
End If
