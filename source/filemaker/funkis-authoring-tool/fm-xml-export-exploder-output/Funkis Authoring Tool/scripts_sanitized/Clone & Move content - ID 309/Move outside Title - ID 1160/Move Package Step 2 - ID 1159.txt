# This script clones and move a package to a new title
# Called from a card window

Set Variable [ $title_id[1] ; FAT Title::gTitle ]
Perform Script [ From list ; "Close card Window" ]

Freeze Window
Set Error Capture [ ON ]

Set Variable [ $old_tt_id[1] ; FAT Title::PrimaryKey ]
Set Variable [ $package_id[1] ; FAT Package::PrimaryKey ]

If [ $old_tt_id = $title_id ]
	Go to Layout [ Layout: "FAT Title" ; Animation: None ]
	Show Custom Dialog [ Title: "Same title" ; Message: "You are trying to move the package to the same title as it is already located. " ; Default Button: "OK" ]
	Exit Script []
Else
	Set Variable [ $old_localisation_code[1] ; FAT Title::Original localisation code ]
	Go to Layout [ Layout: "TABLE FAT Title Alt" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $new_localisation_code[1] ; FAT Title Alt::Original localisation code ]
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	If [ $old_localisation_code ≠ $new_localisation_code ]
		# Cannot change localisation code when moving
		Show Custom Dialog [ Title: "Localisation codes don't match" ; Message: "You are trying to move the package with localisation code \"" & $old_localisation_code & "\" to a title with  localisation code \"" & $new_localisation_code & "\". ¶¶This is not allowed." ; Default Button: "OK" ]
		Go to Layout [ Layout: "FAT Title" ; Animation: None ]
		Exit Script []
	End If
End If

Set Variable [ $package_type_id[1] ; FAT Package::Package type id ]
# Start checking that all blocks and modifiers exist
// Perform Script [ From list ; "Check title compatibility" ; Parameter: SetProperty ( "title_id"; $title_id ) &
				SetProperty ( "old_tt_id"; $old_tt_id ) &
				SetProperty ( "package_type_id"; $package_type_id ) &
				SetProperty ( "package_id"; $package_id ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

Set Variable [ $result[1] ; Get(ScriptResult) ]

If [ $result ≠ "" ]
	Go to Layout [ Layout: "FAT Title" ; Animation: None ]
	Show Custom Dialog [ Title: "Cannot move package" ; Message: $result ; Default Button: "OK" ]
	Exit Script [ $result ]
End If

# Start cloning and moving
Set Field [ FAT Title::gCloneItemInfo ; "" ]
Set Field [ FAT Title::gCloneLevel ; "Package" ]
Set Field [ FAT Title::gCloneItemId ; $package_id ]
Set Field [ FAT Title::gCloneItemParentId ; $old_tt_id ]
Set Field [ FAT Title::gCloneDefaultSettings ; "IOs¶Links¶Lists¶Tags¶Score categories" ]

Set Variable [ $isMovedPackage[1] ; True ]

Commit Records/Requests [ With dialog: OFF ]

Perform Script [ From list ; "Clone Content MASTER" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "old_tt_id" ; $old_tt_id ) &
				SetProperty ( "isMovedPackage" ; $isMovedPackage ) ]

Go to Layout [ Layout: "FAT Title" ; Animation: None ]
