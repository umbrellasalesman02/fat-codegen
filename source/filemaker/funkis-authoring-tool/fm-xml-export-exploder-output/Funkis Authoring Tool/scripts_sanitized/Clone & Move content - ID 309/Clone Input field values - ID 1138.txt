# -----------------------------------------------------------------------------
# --------------- IO FIELD VALUE --------------------------------------------------------

Set Error Capture [ ON ]
Go to Layout [ Layout: "TABLE FAT IO Field Value" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	Set Variable [ $foundCount[1] ; Get(foundCount) ]
	Set Variable [ $recordPosn[1] ; 1 ]
	Go to Record/Request/Page [ First ]
	Loop [ Flush: Always ]
		
		# Check if this will interfere with compound id using Field id and IO value id
		
		Duplicate Record/Request
		
		Set Variable [ $recordPosn[1] ; $recordPosn  + 1 ]
		Set Field [ FAT IO Field Value::IO Value id ; $new_io_value_id ]
		
		If [ $isClonedTitle Or PatternCount($clone_settings; "IOs") ]
			# Point to the input field value created earlier in Make an empty copy
			Set Variable [ $old_input_field_id[1] ; FAT IO Field Value::IO Field id ]
			Go to Layout [ Layout: "TABLE FAT IO Field" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			If [ Get(foundCount) = 1 ]
				Set Variable [ $new_input_field_id[1] ; FAT IO Field::PrimaryKey ]
				Go to Layout [ Layout: "TABLE FAT IO Field Value" ; Animation: None ]
				Set Field [ FAT IO Field Value::IO Field id ; $new_input_field_id ]
			End If
		End If
		
		Exit Loop If [ $recordPosn > $foundCount ]
		Go to Layout [ Layout: "TABLE FAT IO Field Value" ; Animation: None ]
		Go to Record/Request/Page [ With dialog: OFF ; $recordPosn ]
	End Loop
End If
