# Called from Title Children

Set Variable [ $old_tt_id[1] ; GetScriptProperty ( "old_tt_id" ) ]
Set Variable [ $new_tt_id[1] ; GetScriptProperty ( "new_tt_id" ) ]
Set Error Capture [ ON ]

# Duplicate Modifiers
Go to Layout [ Layout: "TABLE FAT Modifier" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
If [ Get(foundCount)>0 ]
	Set Variable [ $foundCount[1] ; Get(foundCount) ]
	Set Variable [ $recordPosn[1] ; 1 ]
	Go to Record/Request/Page [ First ]
	Loop [ Flush: Always ]
		Set Variable [ $old_attribute_id[1] ; FAT Modifier::PrimaryKey ]
		Duplicate Record/Request
		Set Variable [ $recordPosn[1] ; $recordPosn  + 1 ]
		Set Field [ FAT Modifier::Title id ; $new_tt_id ]
		Set Field [ FAT Modifier::Old PrimaryKey from clone ; $old_attribute_id ]
		If [ FAT Modifier::Level = "Block" ]
			Set Variable [ $old_block_id[1] ; FAT Modifier::Parent id ]
			Go to Layout [ Layout: "TABLE FAT Block" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			Set Variable [ $new_block_id[1] ; FAT Block::PrimaryKey ]
			Go to Layout [ Layout: "TABLE FAT Modifier" ; Animation: None ]
			Set Field [ FAT Modifier::Parent id ; $new_block_id ]
		End If
		Exit Loop If [ $recordPosn > $foundCount ]
		Go to Record/Request/Page [ With dialog: OFF ; $recordPosn ]
	End Loop
End If
