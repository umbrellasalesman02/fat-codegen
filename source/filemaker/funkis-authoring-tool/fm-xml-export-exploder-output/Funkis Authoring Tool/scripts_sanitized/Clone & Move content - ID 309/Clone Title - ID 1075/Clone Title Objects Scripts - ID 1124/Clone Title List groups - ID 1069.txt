# Called from Title Objects

Set Variable [ $old_tt_id[1] ; GetScriptProperty ( "old_tt_id" ) ]
Set Variable [ $new_tt_id[1] ; GetScriptProperty ( "new_tt_id" ) ]
Set Error Capture [ ON ]

# Duplicate Blocks - preparing for page items to link via Old PrimaryKey from clone
Go to Layout [ Layout: "TABLE FAT List of Lists" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
If [ Get(foundCount)>0 ]
	Set Variable [ $foundCount[1] ; Get(foundCount) ]
	Set Variable [ $recordPosn[1] ; 1 ]
	Go to Record/Request/Page [ First ]
	Loop [ Flush: Always ]
		Set Variable [ $old_listgroup_id[1] ; FAT List of Lists @ List::PrimaryKey ]
		Duplicate Record/Request
		Set Variable [ $recordPosn[1] ; $recordPosn  + 1 ]
		Set Field [ FAT List of Lists @ List::Title id ; $new_tt_id ]
		Set Field [ FAT List of Lists @ List::Old PrimaryKey from clone ; $old_listgroup_id ]
		Exit Loop If [ $recordPosn > $foundCount ]
		Go to Record/Request/Page [ With dialog: OFF ; $recordPosn ]
	End Loop
End If
