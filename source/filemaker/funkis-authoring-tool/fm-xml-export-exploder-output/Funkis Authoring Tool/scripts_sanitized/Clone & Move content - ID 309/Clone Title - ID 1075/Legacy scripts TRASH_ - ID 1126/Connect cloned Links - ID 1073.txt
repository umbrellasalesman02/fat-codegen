# Called from Clone Title Objects
# After creating Links - go back to List items and swop old Links to new Links
# Links can occure in BOTH List items and Modifier values

Show Custom Dialog [ Title: "Legacy script" ; Message: "This dialog box should not appear when duplicating. Please contact Jens with information: \"Connect cloned Links\"" ; Default Button: "OK" ]

Set Error Capture [ ON ]
Set Variable [ $old_tt_id[1] ; GetScriptProperty ( "old_tt_id" ) ]
Set Variable [ $new_tt_id[1] ; GetScriptProperty ( "new_tt_id" ) ]

# ------------ LIST ITEM --------------------------------------------------
Go to Layout [ Layout: "TABLE FAT List item" ; Animation: None ]
# Find all list items in Title with valuetype = "Link"
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	Loop [ Flush: Always ]
		Set Variable [ $old_link_id[1] ; FAT List item::Value 1 ]
		
		If [ $old_link_id ≠ "" ]
			Go to Layout [ Layout: "TABLE FAT Link" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			Set Variable [ $new_link_id[1] ; FAT Link::PrimaryKey ]
			Go to Layout [ Layout: "TABLE FAT List item" ; Animation: None ]
			Set Field [ FAT List item::Value 1 ; $new_link_id ]
		End If
		
		Go to Layout [ Layout: "TABLE FAT List item" ; Animation: None ]
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
End If

# ------------ MODIFIER VALUE --------------------------------------------------
Go to Layout [ Layout: "TABLE FAT ModifierValue" ; Animation: None ]

# Find all modifier values in Title with valuetype = "Link"
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	Loop [ Flush: Always ]
		Set Variable [ $old_link_id[1] ; FAT Modifier Value::Value ]
		
		If [ $old_link_id ≠ "" ]
			Go to Layout [ Layout: "TABLE FAT Link" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			Set Variable [ $new_link_id[1] ; FAT Link::PrimaryKey ]
			Go to Layout [ Layout: "TABLE FAT ModifierValue" ; Animation: None ]
			Set Field [ FAT Modifier Value::Value ; $new_link_id ]
		End If
		
		Go to Layout [ Layout: "TABLE FAT List item" ; Animation: None ]
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
End If
