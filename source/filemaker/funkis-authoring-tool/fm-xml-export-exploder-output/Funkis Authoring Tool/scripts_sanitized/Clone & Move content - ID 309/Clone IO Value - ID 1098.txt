# This script is called by Clone content SRV  for each item that can have an IO value¨

Set Variable [ $isClonedTitle[1] ; GetAsBoolean(GetScriptProperty ( "isClonedTitle")) ]
Set Variable [ $isMovedPackage[1] ; GetAsBoolean(GetScriptProperty ( "isMovedPackage" )) ]

Set Variable [ $clone_level[1] ; GetScriptProperty ( "clone_level" ) ]
Set Variable [ $clone_settings[1] ; GetScriptProperty ( "clone_settings" ) ]
Set Variable [ $suffix[1] ; GetScriptProperty ( "suffix" ) ]
Set Variable [ $object_group_id[1] ; GetScriptProperty ( "object_group_id" ) ]
Set Variable [ $old_tt_id[1] ; GetScriptProperty ( "old_tt_id") ]
Set Variable [ $title_id[1] ; GetScriptProperty ( "title_id") ]
Set Variable [ $old_item_id[1] ; GetScriptProperty ( "old_item_id") ]
Set Variable [ $new_item_id[1] ; GetScriptProperty ( "new_item_id") ]

Set Error Capture [ ON ]
Go to Layout [ Layout: "TABLE FAT IO Value @ Page item" ; Animation: None ]

# Find IO values for old item
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	Set Variable [ $foundCount[1] ; Get(foundCount) ]
	Set Variable [ $recordPosn[1] ; 1 ]
	Go to Record/Request/Page [ First ]
	Loop [ Flush: Always ]
		Set Variable [ $old_io_value_id[1] ; FAT IO Value @ Page item::PrimaryKey ]
		Set Variable [ $numOfFieldValues[1] ; Count ( FAT IO Field Value @ IO Value PI::PrimaryKey ) ]
		
		Duplicate Record/Request
		
		Set Variable [ $io_value_id[1] ; FAT IO Value @ Page item::PrimaryKey ]
		Set Field [ FAT IO Value @ Page item::Parent id ; $new_item_id ]
		
		Set Variable [ $recordPosn[1] ; $recordPosn  + 1 ]
		
		If [ $isClonedTitle Or PatternCount($clone_settings; "IOs") ]
			Set Variable [ $old_io_id[1] ; FAT IO Value @ Page item::IO id ]
			Go to Layout [ Layout: "TABLE FAT IO" ; Animation: None ]
			Perform Find [ ⚠️ RESTORE ⚠️ ]
			If [ Get(foundCount) = 0 ]
				# No oldPrimaryKey found. Create a new IO and add it to the IO value
				Perform Script [ From list ; "Clone IO SRV" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
								SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
								SetProperty ( "clone_level" ; $clone_level ) &
								SetProperty ( "clone_settings" ; $clone_settings ) &
								SetProperty ( "suffix" ; $suffix ) &
								SetProperty ( "object_group_id" ; $object_group_id ) &
								SetProperty ( "old_tt_id" ; $old_tt_id ) &
								SetProperty ( "title_id" ; $title_id ) &
								SetProperty ( "old_io_id" ; $old_io_id ) ]
				Set Variable [ $new_io_id[1] ; Get(ScriptResult) ]
			Else
				Set Variable [ $new_io_id[1] ; FAT IO::PrimaryKey ]
			End If
			Go to Layout [ Layout: "TABLE FAT IO Value @ Page item" ; Animation: None ]
			Set Field [ FAT IO Value @ Page item::IO id ; $new_io_id ]
			Commit Records/Requests [ With dialog: OFF ]
			# Clone Field List Sync Id here
			
			
		End If
		
		# Clone Input field values, IO Value conditions and IO Value SortOrder here
		Set Variable [ $io_id[1] ; FAT IO Value @ Page item::IO id ]
		If [ $numOfFieldValues > 0 ]
			Perform Script [ From list ; "Clone Input field values" ; Parameter: SetProperty ( "isClonedTitle" ; $isClonedTitle ) &
							SetProperty ( "isMovedPackage" ; $isMovedPackage ) &
							SetProperty ( "clone_level" ; $clone_level ) &
							SetProperty ( "clone_settings" ; $clone_settings ) &
							SetProperty ( "suffix" ; $suffix ) &
							SetProperty ( "title_id" ; $title_id ) &
							SetProperty ( "old_io_value_id" ; $old_io_value_id ) &
							SetProperty ( "new_io_value_id" ; $io_value_id ) &
							SetProperty ( "io_id" ; $io_id ) ]
		End If
		
		Go to Layout [ Layout: "TABLE FAT IO Value @ Page item" ; Animation: None ]
		Exit Loop If [ $recordPosn > $foundCount ]
		Go to Record/Request/Page [ With dialog: OFF ; $recordPosn ]
	End Loop
End If
