# This script is used for asynchronous machine translations. 
  When sending a text to this kind of api, FAT saves the request id in the field "aSync translation id". 

# The script will be called repeateadily until all texts are translated.
# Called from script: Async Get Texts Button

Set Error Capture [ ON ]

# Stage  Name                             Text status HT              Text status MT
  1.     Language added                   - (text does not exist)     - (text does not exist)
  2.     Text prepared for translation    Not translated              Not translate
  3.     Original text sent	              Not translated               Sent to API + async translation id:¶[id_from_api] (async only)
  4.     Translated text imported	      Not reviewed                 Not reviewed
  5.	   Translation in review           In progress                  In progress
  6.     Translation approved             Approved                    Approved

Set Variable [ $title_id[1] ; GetScriptProperty ("title_id") ]
Set Variable [ $package_id[1] ; GetScriptProperty ("package_id") ]
Set Variable [ $localisation_id[1] ; GetScriptProperty ("localisation_id") ]

Set Variable [ $original_localisation_code[1] ; GetScriptProperty ( "original_localisation_code" ) ]
Set Variable [ $target_localisation_code[1] ; GetScriptProperty ( "target_localisation_code" ) ]

# THe translation API
Set Variable [ $translation_api[1] ; GetScriptProperty ( "translation_api" ) ]

# Get all info for the translation API
Go to Layout [ Layout: "TABLE Translation API" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
Set Variable [ $get_api_url[1] ; Translation API @ Title::Get translation URL ]
Set Variable [ $api_key[1] ; Translation API @ Title::API PrivateKey ]
Set Variable [ $translation_api_name[1] ; Translation API @ Title::API Name ]
Set Variable [ $translation_api_error_message[1] ; Translation API @ Title::API Error Message ]

Go to Layout [ Layout: "TABLE FAT Text Translated" ; Animation: None ]

# Find all texts in Title with the target language and country code with status "Sent to API" which means it has an unfetched translation
Set Variable [ $status[1] ; "Sent to API" ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

If [ Get(foundCount)>0 ]
	# Sort texts using recordNumberReference if exist
	Sort Records [ With dialog: OFF ; ⚠️ PARAMETER "Restore" NOT PARSED ⚠️ ; ⚠️ PARAMETER "SortSpecification" NOT PARSED ⚠️ ]
	Go to Record/Request/Page [ First ]
	
	Set Variable [ $num_of_successful_translations[1] ; 0 ]
	Set Variable [ $num_of_waiting_translations[1] ; 0 ]
	Set Variable [ $num_of_failed_translations[1] ; 0 ]
	Loop [ Flush: Always ]
		Set Variable [ $translation_id[1] ; FAT Text translated::aSync translation id ]
		
		# ------------     TRANSLATE     ------------------------------------------------
		Perform Script [ From list ; "Get text from API" ; Parameter: SetProperty( "translation_id" ; $translation_id ) &
						SetProperty( "translation_api" ; $translation_api ) &
						SetProperty( "get_api_url"; $get_api_url) &
						SetProperty( "api_key"; $api_key) ]
		
		Set Variable [ $result[1] ; Get(ScriptResult) ]
		# Reset the asynchronous translation id regardless if the fetch suceeded or not
		If [ $result = "--- NOT TRANSLATED YET ---" ]
			# Wait and see if the translation may be finalised later
			Set Variable [ $num_of_waiting_translations[1] ; $num_of_waiting_translations + 1 ]
		Else If [ $result = $translation_api_error_message Or IsEmpty ( $result ) ]
			# If response includes a pre-defined response i.e. translation not available anymore - reset textfield to "Not translated" - then it can be re-translated
			Set Field [ FAT Text translated::Status ; "Not translated" ]
			Set Field [ FAT Text translated::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; "Not translated" ]
			Set Field [ FAT Text translated::aSync translation id ; "" ]
			Set Variable [ $num_of_failed_translations[1] ; $num_of_failed_translations + 1 ]
		Else
			# Result is the translated text
			Set Variable [ $num_of_successful_translations[1] ; $num_of_successful_translations + 1 ]
			Set Field [ FAT Text translated::Text ; $result ]
			Set Field [ FAT Text translated::Status ; "Not reviewed" ]
			Set Field [ FAT Text translated::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; "Not reviewed" ]
			Set Field [ FAT Text translated::aSync translation id ; "" ]
		End If
		
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
End If

If [ $num_of_failed_translations = 0 And $num_of_waiting_translations = 0 ]
	Exit Script [ "Successfully fetched all " & $num_of_successful_translations & " texts." ]
Else
	Exit Script [ "Successfully fetched " & $num_of_successful_translations & " texts." & "Failed translations: " & $num_of_failed_translations & ", Not ready translations " & $num_of_waiting_translations ]
End If
