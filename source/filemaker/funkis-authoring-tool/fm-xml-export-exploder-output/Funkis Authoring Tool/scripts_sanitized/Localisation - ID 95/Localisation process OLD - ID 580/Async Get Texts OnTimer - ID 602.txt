# This script is called from an OnTimer script initiadet in Translation MASTER script
# It is used for Asynchronous translation APIs
# Stage  Name                             Text status HT              Text status MT
  1.     Language added                   - (text does not exist)     - (text does not exist)
  2.     Text prepared for translation    Not translated              Not translate
  3.     Original text sent	              Not translated               Sent to API + async translation id:¶[id_from_api] (async only)
  4.     Translated text imported	      Not reviewed                 Not reviewed
  5.	   Translation in review           In progress                  In progress
  6.     Translation approved             Approved                    Approved

Freeze Window

Set Variable [ $$async_calls[1] ; $$async_calls + 1 ]

Set Variable [ $localisation_id[1] ; $$localisation_id ]
Go to Layout [ Layout: "TABLE FAT Localisation @ Package" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

Set Variable [ $title_id[1] ; FAT Localisation @ Package::Title id ]
Set Variable [ $package_id[1] ; FAT Localisation @ Package::Package id ]

Set Variable [ $original_localisation_code[1] ; FAT Localisation @ Package::Original Localisation code ]
Set Variable [ $target_localisation_code[1] ; FAT Localisation @ Package::Localisation code ]

# THe translation API
Set Variable [ $translation_api[1] ; FAT Localisation @ Package::Translation API ]

Perform Script [ From list ; "Get all texts from API" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "package_id" ; $package_id ) &
				SetProperty( "original_localisation_code" ; $original_localisation_code) &
				SetProperty( "target_localisation_code" ; $target_localisation_code) &
				SetProperty( "translation_api"; $translation_api) ]
// Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

Set Variable [ $async_translation_result[1] ; Get(ScriptResult) ]
If [ $async_translation_result = "success" ]
	# Remove the recurrent script
	Install OnTimer Script
	# Change stage to 4
	Perform Script [ From list ; "Change Package localisation stage" ; Parameter: SetProperty ( "localisation_id" ; $localisation_id ) &
					SetProperty ( "stage" ; 4 ) ]
	# Add event to log
	Set Variable [ $event_text[1] ; "Localisation set to stage 4." ]
	Perform Script [ From list ; "Add to localisation log" ; Parameter: SetProperty ( "localisation_id" ; $localisation_id ) & 
					SetProperty ( "event_text" ; $event_text ) ]
	
	Go to Layout [ Layout: "FAT Package Settings" ; Animation: None ]
End If


If [ $$async_calls >5 ]
	# Remove the OnTimer script
	Install OnTimer Script
	Set Variable [ $event_text[1] ; "Failed to get texts from API. " & $async_translation_result ]
	Perform Script [ From list ; "Add to localisation log" ; Parameter: SetProperty ( "localisation_id" ; $localisation_id ) & 
					SetProperty ( "event_text" ; $event_text ) ]
End If
