# This translation flow is designed for Title Translations
# Script is called from Trasnlate button in Title Localisation Card Window when translation method = "Machine"

# Stage  Name                             Text status HT              Text status MT
  1.     Language added                   - (text does not exist)     - (text does not exist)
  2.     Text prepared for translation    Not translated              Not translated
  3.     Original text sent	              Not translated               Sent to API + async translation id:¶[id_from_api] (async only)
  4.     Translated text imported	      Not reviewed                 Not reviewed
  5.	   Translation in review           In progress                  In progress
  6.     Translation approved             Approved                    Approved

# For Machine translation - localisation will be in STAGE 4 after this script if STAGE < 4 when triggered

Freeze Window
Set Error Capture [ ON ]
Commit Records/Requests [ With dialog: OFF ]

# ----------------- Intitiate all variables from Localisation that will be used in process ----------------------------------
Set Variable [ $title_id[1] ; FAT Title::PrimaryKey ]
Set Variable [ $title_localisation_id[1] ; FAT Localisation::PrimaryKey ]
Set Variable [ $localisation_stage[1] ; FAT Localisation::Localisation stage ]
Set Variable [ $last_sorted_timestamp[1] ; FAT Title::TextInContextualOrderTimeStamp ]
Set Variable [ $last_localised_textfield_creation_timestamp[1] ; Case ( 
				
				   IsEmpty ( FAT Localisation::TextfieldsCreationTimestamp );
				
				   GetAsTimeStamp ( "2000-01-01" );
				
				   FAT Localisation::TextfieldsCreationTimestamp
				
				 ) ]
Set Variable [ $original_localisation_code[1] ; FAT Localisation::Original Localisation code ]
Set Variable [ $target_localisation_code[1] ; FAT Localisation::Localisation code ]

# -------------- Logging start time for localisation -----------------
Set Variable [ $event_text[1] ; "Machine localisation started." ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

# ------------- Check if texts needs to be arranged in a new sort order ---------------------------------
# ------------- Returns a boolean AND timestamp when last original text field was created ---------------------------------
// Perform Script [ From list ; "Check if arrange texts needed" ; Parameter: SetProperty ( "title_id" ; $title_id ) & 
				SetProperty ( "original_localisation_code" ; $original_localisation_code ) & 
				SetProperty ( "last_sorted_timestamp" ; $last_sorted_timestamp ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
Set Variable [ $new_arrange_needed[1] ; GetAsBoolean ( GetResultProperty ( "new_arrange_needed" )) ]
Set Variable [ $last_original_textfield_creation_timestamp[1] ; GetAsTimeStamp ( GetResultProperty ( "last_original_textfield_creation_timestamp" )) ]

# ------------- If texts needs to be arranged in a new sort order - Arrange text ---------------------------------
If [ $new_arrange_needed ]
	
	Show Custom Dialog [ Title: "Arrange text in contextual order" ; Message: "Before starting the translation, texts needs to be arranged in the right contextual order.¶¶This might take some time." ; Default Button: "Arrange" ; Button 2: "Skip" ; Button 3: "Stop" ]
	If [ Get(lastMessageChoice) = 3 ]
		# User pressed Stop - Add event to log
		Set Variable [ $event_text[1] ; "Texts needed new arrangement. User stopped. Last text created " & $last_original_textfield_creation_timestamp & "." ]
		Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
		Exit Script []
	Else If [ Get(lastMessageChoice) = 1 ]
		
		// Perform Script [ From list ; "Arrange text in order Title" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
						SetProperty ( "title_localisation_id" ; $title_localisation_id ) ]
		Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
		
		Set Variable [ $last_sorted_timestamp[1] ; Get(currentHostTimeStamp) ]
		If [ Not PatternCount ( Get(ScriptResult) ; "error" ) ]
			# Title and Package texts now are in contextual order
			# If no errors - save the timestamp when finding new texts
			Set Field [ FAT Title::TextInContextualOrderTimeStamp ; $last_sorted_timestamp ]
			Set Variable [ $event_text[1] ; "Texts were successfully arranged in contextual order." ]
			Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
		Else
			# Something went wrong when arranging texts. Maybe a record is busy.
			Go to Layout [ Layout: <original layout> ; Animation: None ]
			Show Custom Dialog [ Title: "Error when sorting text" ; Message: "Something went wrong when sorting the original texts in contextual order.¶Probably there's someone else working with the title right now.¶¶Do you want to stop the process or continue?" ; Default Button: "Stop" ; Button 2: "Continue" ]
			If [ Get(lastMessageChoice) = 1 ]
				Set Variable [ $event_text[1] ; "User stopped because an error occured when arranging texts in contextual order. " & Get(ScriptResult) ]
				Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
				Exit Script []
			End If
		End If
	Else If [ Get(lastMessageChoice) = 2 ]
		# User pressed Skip - Add event to log
		Set Variable [ $event_text[1] ; "Texts needed new arrangement but user skipped. Last text created " & $last_original_textfield_creation_timestamp & "." ]
		Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
	End If
End If

# ---------------------- Stage 1 - Localise text fields - Duplicate new texts ---------------------------------

# New strategy opening a separate FAT Text file to increase performance
# For 40 text fields time reduced from 9 sec to 3 sec using FM Server 20.3.1 with Loop Flush: Defer setting
# With previous server version with Loop Flush: Always - time was probably longer. 
// Open File [ Open hidden: ON ; "FAT Text" ]
// Perform Script [ From list ; "Localise Text fields" ; File: "FAT Text" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "original_localisation_code" ; $original_localisation_code ) &
				SetProperty ( "target_localisation_code" ; $target_localisation_code ) &
				SetProperty ( "last_localised_textfield_creation_timestamp" ; $last_localised_textfield_creation_timestamp ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
Set Variable [ $num_of_created[1] ; GetResultProperty ( "num_of_created") ]
Set Variable [ $num_of_errors[1] ; GetResultProperty ( "num_of_errors") ]
Set Variable [ $last_localised_textfield_creation_timestamp[1] ; GetAsTimeStamp (GetResultProperty ( "last_localised_textfield_creation_timestamp")) ]
// Close File [ "FAT Text" ]

If [ $num_of_errors > 0 ]
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	Show Custom Dialog [ Title: "Error when creating new text fields for localisation" ; Message: "Something went wrong when creating new localised text fields.¶Probably there's someone else working with the title right now.¶¶Do you want to stop the process or continue?" ; Default Button: "Stop" ; Button 2: "Continue" ]
	If [ Get(lastMessageChoice) = 1 ]
		Set Variable [ $event_text[1] ; "User stopped because error when creating new localised text fields. " ]
		Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
		Exit Script []
	End If
	Set Variable [ $event_text[1] ; "Error when creating new localised text fields but user continued process. " ]
Else
	Set Variable [ $event_text[1] ; "Successfully created " & $num_of_created & " new localised text fields." ]
	# Update Localisation with date when localised text fields were created.
	Set Field [ FAT Localisation::TextfieldsCreationTimestamp ; $last_localised_textfield_creation_timestamp ]
	# The commit below is needed for allowing PSOS to make changes
	Commit Records/Requests [ With dialog: OFF ]
End If
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

# ------------------ Stage 2 - Send text to translation and put result in textfields --------------------------
Set Variable [ $localisation_stage[1] ; 2 ]
// Perform Script [ From list ; "Change Title localisation stage" ; Parameter: SetProperty ( "title_localisation_id" ; $title_localisation_id ) &
				SetProperty ( "localisation_stage" ; $localisation_stage ) &
				SetProperty ( "force_stage" ; "" ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

# ----------------- FUNKIS CLOUD MACHINE TRANSLATION ---------------------------------------------
// Perform Script [ From list ; "FC Machine translation" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "localisation_id" ; $title_localisation_id ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

Set Variable [ $num_of_translated_texts[1] ; GetResultProperty ( "num_of_translated_texts") ]
Set Variable [ $num_of_untranslated_texts[1] ; GetResultProperty ( "num_of_untranslated_texts") ]
Set Variable [ $api_name[1] ; GetResultProperty ( "api_name") ]
Set Variable [ $result[1] ; GetResultProperty ( "result") ]

If [ PatternCount ( $result ; "Error" ) ]
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	Show Custom Dialog [ Title: "Error when translating" ; Message: $result ; Default Button: "OK" ]
	Set Variable [ $event_text[1] ; "Error when sending or retrieving new texts from translation API " & $api_name & ". " & $result & "." ]
	Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
	Exit Script []
End If


# ---------------------- Stage 4 - Change stage on Title and Package localisations --------------------------------------------
Set Variable [ $localisation_stage[1] ; 4 ]

Set Variable [ $event_text[1] ; Case ( $num_of_untranslated_texts=0; 
				
				  "Successfully translated " & $num_of_translated_texts & " texts using " & $api_name & ".";
				
				  "Successfully translated " & $num_of_translated_texts & " texts using " & $api_name & ". Could not translate " &
				    $num_of_untranslated_texts & " texts."
				
				) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

// Perform Script [ From list ; "Change Title localisation stage" ; Parameter: SetProperty ( "title_localisation_id" ; $title_localisation_id ) &
				SetProperty ( "localisation_stage" ; $localisation_stage ) &
				SetProperty ( "force_stage" ; "" ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]


# ---------------------- Update localisation logs with status --------------------------------------------

Go to Layout [ Layout: <original layout> ; Animation: None ]
// Perform Script [ From list ; "Log translation status CL" ; Parameter: SetProperty ( "caller" ; "Title localisation" ) &
				SetProperty ( "localisation_id" ; $title_localisation_id ) &
				SetProperty ( "localisation_stage" ; $localisation_stage ) &
				SetProperty ( "status_type" ; "status" ) ]
Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]

# Make sure all PSOS changes are viewed in window
Refresh Window
