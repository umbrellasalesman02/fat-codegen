Set Variable [ $title_id[1] ; GetScriptProperty ( "title_id" ) ]
Set Variable [ $title_localisation_id[1] ; GetScriptProperty ( "title_localisation_id" ) ]
Go to Layout [ Layout: "TABLE FAT Localisation" ; Animation: None ]

# Find all Package localisations based on Title localisation
Go to Layout [ Layout: "TABLE FAT Localisation" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]

Freeze Window

If [ Get(foundCount) >0 ]
	Loop [ Flush: Defer ]
		
		Perform Script [ From list ; "Arrange text in order Package" ; Parameter: SetProperty ( "title_id" ; $title_id) &
						SetProperty ( "package_id" ; FAT Localisation::Package id ) &
						SetProperty ( "localisation_id" ; FAT Localisation::PrimaryKey) &
						SetProperty ( "localisation_code" ; FAT Localisation::Original Localisation code) ]
		Set Variable [ $result[1] ; $result  & Get(ScriptResult) & ¶ ]
		# If no error when arranging texts - set timestamp when arranged
		If [ Not PatternCount ( Get(ScriptResult) ; "error" ) ]
			Set Field [ FAT Package @ Localisation::TextInContextualOrderTimeStamp ; Get(currentHostTimeStamp) ]
		End If
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	Set Field [ FAT Title::TextInContextualOrderTimeStamp ; Get(currentHostTimeStamp) ]
End If

Exit Script [ $result ]
