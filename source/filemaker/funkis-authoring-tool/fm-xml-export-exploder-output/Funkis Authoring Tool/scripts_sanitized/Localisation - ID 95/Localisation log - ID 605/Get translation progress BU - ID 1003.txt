# This script is called from both translation scripts and from review scripts

# Updated 2023-11-19 - Package progress ONLY include Package texts and Shared Titles texts are managed in a separate localisation

Set Error Capture [ ON ]

# Stage  Name                             Text status HT              Text status MT
  1.     Language added                   - (text does not exist)     - (text does not exist)
  2.     Text prepared for translation    Not translated              Not translate
  3.     Original text sent	              Not translated               Sent to API + async translation id:¶[id_from_api] (async only)
  4.     Translated text imported	      Not reviewed                 Not reviewed
  5.	   Translation in review           In progress                  In progress
  6.     Translation approved             Approved                    Approved

# Status type = "progress" or "status" - decides what to report back
Set Variable [ $status_type[1] ; GetScriptProperty ( "status_type" ) ]

# Either send in localisation id
Set Variable [ $localisation_id[1] ; GetScriptProperty ( "localisation_id" ) ]

# Or Title, Package and localisation code - from translation review?
# Since 2023-11-19 onlyt Title progress is shown in Translation review - so no need for package id? 

Set Variable [ $title_id[1] ; GetScriptProperty ( "title_id" ) ]
Set Variable [ $package_id[1] ; GetScriptProperty ( "package_id" ) ]
Set Variable [ $localisation_code[1] ; GetScriptProperty ( "localisation_code" ) ]

Go to Layout [ Layout: "TABLE FAT Localisation @ Package" ; Animation: None ]
If [ $localisation_id = "" ]
	If [ $package_id ≠ "" ]
		Perform Find [ ⚠️ RESTORE ⚠️ ]
		Set Variable [ $localisation_id[1] ; FAT Localisation @ Package::PrimaryKey ]
	Else
		# Show all localisation for this Title
		Perform Find [ ⚠️ RESTORE ⚠️ ]
	End If
Else
	# Find the localisation
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $title_id[1] ; FAT Localisation @ Package::Title id ]
	Set Variable [ $package_id[1] ; FAT Localisation @ Package::Package id ]
	Set Variable [ $localisation_code[1] ; FAT Localisation @ Package::Localisation code ]
	Set Variable [ $shared_text_localisation_id[1] ; FAT Localisation @ Package::Shared text localisation id ]
	Set Variable [ $shared_text_localisation[1] ; FAT Localisation @ Package::Shared text localisation ]
End If

# Update all summary fields
Go to Layout [ Layout: "TABLE FAT Text Translated" ; Animation: None ]

# Text status can be:
  1. Not translated
  2. Sent to API (only asynchronous machine translation)
  3. Not reviewed
  4. In progress
  5. Approved

If [ Not IsEmpty ( $package_id ) ]
	# Now Package only texts are logged
	Set Variable [ $status[1] ; "Not translated" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $not_translated[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Sent to API" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $sent_to_api[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Not reviewed" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $not_reviewed[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "In progress" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $in_progress[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Approved" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $approved[1] ; Get(foundCount) ]
	
Else If [ $shared_text_localisation ]
	
	# Only Title shared texts in this report
	Set Variable [ $status[1] ; "Not translated" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $not_translated[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Sent to API" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $sent_to_api[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Not reviewed" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $not_reviewed[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "In progress" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $in_progress[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Approved" ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $approved[1] ; Get(foundCount) ]
	
Else
	# All texts
	# If looking at all texts in the whiole Title - from Translation review
	  Change 2023-11-16: Only original texts with texts (not empty) are included but it took TOO long time to search
	
	Set Variable [ $status[1] ; "Not translated" ]
	// Perform Find [ ⚠️ RESTORE ⚠️ ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	// Constrain Found Set [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $not_translated[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Sent to API" ]
	// Perform Find [ ⚠️ RESTORE ⚠️ ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	// Constrain Found Set [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $sent_to_api[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Not reviewed" ]
	// Perform Find [ ⚠️ RESTORE ⚠️ ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	// Constrain Found Set [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $not_reviewed[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "In progress" ]
	// Perform Find [ ⚠️ RESTORE ⚠️ ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	// Constrain Found Set [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $in_progress[1] ; Get(foundCount) ]
	
	Set Variable [ $status[1] ; "Approved" ]
	// Perform Find [ ⚠️ RESTORE ⚠️ ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	// Constrain Found Set [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $approved[1] ; Get(foundCount) ]
	
End If


Go to Layout [ Layout: "TABLE FAT Localisation @ Package" ; Animation: None ]
# Showing all localisations for Title

If [ Not IsEmpty ( $package_id ) Or $shared_text_localisation ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $not_translated ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $sent_to_api ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $not_reviewed ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $in_progress ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $approved ]
Else
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $not_translated ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $sent_to_api ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $not_reviewed ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $in_progress ]
	Replace Field Contents [ FAT Localisation @ Package::🚨🚨🚨 <BROKEN REFERENCE> 🚨🚨🚨 ; Replace with calculation: $approved ]
End If

Commit Records/Requests [ With dialog: OFF ]

If [ $status_type = "progress" ]
	Exit Script [ Round ( 100*$approved/($not_translated + $sent_to_api + $not_reviewed + $in_progress + $approved) ; 0) & "% texts approved." ]
Else If [ $status_type = "status" ]
	Exit Script [ "Not translated: " & $not_translated & ", Sent to API: "& $sent_to_api & ", Not reviewed: " &  $not_reviewed & ", In progress: " & $in_progress & ", Appproved: " & $approved ]
End If
