# Searches all texts in this localisation and updates the text fields trasnaltion status.
# If Title localisation - all Package localisatins are updated too.
# Called from buttons in Localisation settings and in the end of Title MT Localisation MASTER

Freeze Window
Set Error Capture [ ON ]

# Caller script parameter can be:
  "Title localisation"
  "Package localisation"
  

# Status_type script parameter can be:
  "progress" - check translation approval status in %
  "size" - number of text chunks in localisation
  "status" - check number of texts for each text status (see below)

# Text status can be:
  1. Not translated
  2. Sent to API (only asynchronous machine translation)
  3. Not reviewed
  4. In progress
  5. Approved

Set Variable [ $caller[1] ; GetScriptProperty ( "caller" ) ]
Set Variable [ $localisation_id[1] ; GetScriptProperty ( "localisation_id" ) ]
Set Variable [ $localisation_stage[1] ; GetScriptProperty ( "localisation_stage" ) ]
Set Variable [ $status_type[1] ; GetScriptProperty ( "status_type" ) ]

# If not PSOS - this script will not work properly when caller = "Title localisation" since FAT Localisation @ Package table is used for two purposes
If [ isPSOS ]
	Perform Script [ From list ; "Log translation status SRV" ; Parameter: SetProperty ( "localisation_id" ; $localisation_id ) &
					SetProperty ( "localisation_stage" ; $localisation_stage ) &
					SetProperty ( "status_type" ; $status_type ) ]
Else
	// Perform Script [ From list ; "Log translation status SRV" ; Parameter: SetProperty ( "localisation_id" ; $localisation_id ) &
					SetProperty ( "localisation_stage" ; $localisation_stage ) &
					SetProperty ( "status_type" ; $status_type ) ]
	Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
End If

If [ $caller = "Title localisation" ]
	# If Title localisation - add to all childrens event log
	# Always use the Localisation @ Package layout when doing children operations
	Go to Layout [ Layout: "TABLE FAT Localisation @ Package" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	If [ Get(foundCount)>0 ]
		Loop [ Flush: Always ]
			Set Variable [ $localisation_id[1] ; FAT Localisation @ Package::PrimaryKey ]
			Set Variable [ $localisation_stage[1] ;  FAT Localisation @ Package::Localisation stage ]
			If [ isPSOS ]
				Perform Script [ From list ; "Log translation status SRV" ; Parameter: SetProperty ( "localisation_id" ; $localisation_id ) & 
								SetProperty ( "localisation_stage" ; $localisation_stage ) &
								SetProperty ( "status_type" ; $status_type ) ]
			Else
				Perform Script on Server [ From list ; ⚠️ PARAMETER "Parameter" NOT PARSED ⚠️ ; Wait for completion: ON ]
			End If
			Go to Record/Request/Page [ Next ; Exit after last: ON ]
		End Loop
	End If
End If

Go to Layout [ Layout: <original layout> ; Animation: None ]

If [ Not isPSOS ]
	Go to Object [ "Event log" ; Repetition: 1 ]
	Refresh Portal [ ⚠️ PARAMETER "Object" NOT PARSED ⚠️ ]
End If
