# Called from both Title Settings and User adminsitration layout
Set Variable [ $user_id[1] ; Get(scriptParameter) ]
Show Custom Dialog [ Title: "Reset password" ; Message: "If pressing OK, the password for this user will be resetted.
				
				Are you sure you want to reset the password for this user?" ; Default Button: "OK" ; Button 2: "Cancel" ; Input 1: $password ; Label 1: "New password" ]
If [ Get(lastMessageChoice) = 1 ]
	Reset Account Password [ Name: $user_id ; ⚠️ PARAMETER "Password" NOT PARSED ⚠️ ; Password: ON ]
	If [ Get(lastError) = 212 ]
		# Account has been auto-deleted - create again
		Add Account [ Name: $user_id ; ⚠️ PARAMETER "Password" NOT PARSED ⚠️ ; ⚠️ PARAMETER "PrivilegeSetReference" NOT PARSED ⚠️ ; Expire password: ON ]
	Else If [ Get(lastError)≠ 0 ]
		Show Custom Dialog [ Title: "Error message" ; Message: "Error: " & Get(lastError) & "¶" &
						"Details: " & Get(lastErrorDetail) & "¶" &
						"Location: " & Get(LastErrorLocation) ; Default Button: "OK" ]
		Exit Script []
	End If
	Open URL [ In external browser: OFF ; With dialog: OFF ; ⚠️ PARAMETER "URL" NOT PARSED ⚠️ ]
End If
