Freeze Window
Perform Script [ From list ; "Save Active Layout" ]
Set Variable [ $level[1] ; Get(scriptParameter) ]
If [ $level = "Title" ]
	Go to Related Record [ Table: FAT Title ; Layout: "FAT Title" ; Show related ]
Else If [ $level = "Package" ]
	Go to Related Record [ Table: FAT Package ; Layout: "FAT Package" ; Show related ]
Else If [ $level = "Chapter" ]
	Go to Related Record [ Table: FAT Chapter ; Layout: "FAT Chapter" ; Show related ]
Else If [ $level = "Page" ]
	Go to Related Record [ Table: FAT Page ; Layout: "FAT Page" ; Show related ]
End If
