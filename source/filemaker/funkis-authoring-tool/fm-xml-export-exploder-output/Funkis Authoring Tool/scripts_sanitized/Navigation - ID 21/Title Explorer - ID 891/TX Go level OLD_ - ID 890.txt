# Only called from small navigation window
Set Variable [ $target_level[1] ; Get(scriptParameter) ]
Perform Script [ From list ; "TX Save Layout" ]
If [ $target_level = "chapter" ]
	Go to Related Record [ Table: FAT Chapter ; Layout: "FAT TX Chapter" ; Show related ]
Else If [ $target_level = "page" ]
	Go to Related Record [ Table: FAT Page ; Layout: "FAT TX Page" ; Show related ]
Else If [ $target_level = "page-item" ]
End If
