Set Variable [ $item_id[1] ; Let ( 
				
				
				  [
				
				   url = $url;
				
				    pos = Position ( url ; "&param=" ; 0 ; 1 );
				
				    id = Right ( url ; Length ( url )- pos - 6 )
				
				
				
				   ];
				
				
				id
				
				 ) ]

Set Variable [ $script_name[1] ; Let ( 
				
				
				  [
				
				   url = $url;
				
				    pos_start = Position ( url ; "script=" ; 0 ; 1 );
				    pos_end = Position ( url ; "&param=" ; 0 ; 1 );
				
				    script = Middle ( url ; pos_start + 7 ; pos_end -  pos_start- 7)
				
				
				
				   ];
				
				
				script
				
				 ) ]

If [ $script_name = "linkTriggeredOpenTitle" ]
	Perform Script [ From list ; "linkTriggeredOpenTitle" ; Parameter: $item_id ]
Else If [ $script_name = "linkTriggeredOpenPackage" ]
	Perform Script [ From list ; "linkTriggeredOpenPackage" ; Parameter: $item_id ]
Else If [ $script_name = "linkTriggeredOpenChapter" ]
	Perform Script [ From list ; "linkTriggeredOpenChapter" ; Parameter: $item_id ]
Else If [ $script_name = "linkTriggeredOpenPage" ]
	Perform Script [ From list ; "linkTriggeredOpenPage" ; Parameter: $item_id ]
