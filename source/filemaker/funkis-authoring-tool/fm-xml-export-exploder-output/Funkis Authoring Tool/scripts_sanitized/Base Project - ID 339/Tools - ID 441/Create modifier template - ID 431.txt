If [ Get(foundCount)>0 ]
	Loop [ Flush: Always ]
		Set Variable [ $level[1] ; FAT Modifier::Level ]
		Set Variable [ $attribute_type[1] ; FAT Modifier::Type ]
		Set Variable [ $attribute_name[1] ; FAT Modifier::Name ]
		Set Variable [ $attribute_default_value[1] ; FAT Modifier::DefaultValue ]
		Set Variable [ $attribute_description[1] ; FAT Modifier::Description ]
		Go to Layout [ Layout: "TABLE ModifierTemplate" ; Animation: None ]
		New Record/Request
		Set Field [ ModifierTemplate @ BlockTemplate::Base Project ; $code_base_id ]
		Set Field [ ModifierTemplate @ BlockTemplate::Name ; $attribute_name ]
		Set Field [ ModifierTemplate @ BlockTemplate::Level ; $level ]
		Set Field [ ModifierTemplate @ BlockTemplate::Type ; $attribute_type ]
		Set Field [ ModifierTemplate @ BlockTemplate::DefaultValue ; $attribute_default_value ]
		Set Field [ ModifierTemplate @ BlockTemplate::Description ; $attribute_description ]
		Go to Layout [ Layout: "TABLE FAT Modifier" ; Animation: None ]
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
End If
