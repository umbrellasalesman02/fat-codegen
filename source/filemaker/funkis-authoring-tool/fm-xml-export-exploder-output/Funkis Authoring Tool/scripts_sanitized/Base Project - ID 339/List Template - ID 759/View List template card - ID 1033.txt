# Called from a button in Title Link Layout
Commit Records/Requests [ With dialog: OFF ]
Set Variable [ $list_template_id[1] ; Get(scriptParameter) ]

Set Variable [ $callLayout[1] ; Case(Get(WindowStyle) = 3; Get(layoutName); "") ]
Perform Script [ From list ; "Open card setting" ; Parameter: SetProperty ("card_window_name" ; "List template" ) &
				SetProperty ("card_window_layout" ; "List Template Card" ) &
				SetProperty ("card_window_width" ; "1140" ) &
				SetProperty ("card_window_height" ; "680" ) &
				SetProperty ("item_id" ; $list_template_id ) &
				SetProperty ("item_type" ; "List template" ) & 
				SetProperty ( "callLayout" ; $callLayout) ]
