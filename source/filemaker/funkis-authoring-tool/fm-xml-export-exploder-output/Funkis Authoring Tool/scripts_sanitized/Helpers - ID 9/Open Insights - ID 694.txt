If [ Platform = "WebDirect" ]
	// Show Custom Dialog [ Title: "Web" ; Message: Platform ; Default Button: "OK" ]
	Set Variable [ $parameters[1] ; "$fat_title_id=" & $fat_title_id & "&" &
					"$funkis_cloud_title_id=" & $funkis_cloud_title_id & "&" &
					"$title_environment=" & $title_environment ]
	If [ $fat_title_id ≠ "" And $funkis_cloud_title_id≠"" ]
		Open URL [ In external browser: OFF ; With dialog: OFF ; ⚠️ PARAMETER "URL" NOT PARSED ⚠️ ]

Else

	If [ $fat_title_id ≠ "" And $funkis_cloud_title_id≠"" ]
		Perform Script [ From list ; "LinkToInsightTitle" ; File: "Funkis Insight" ; Parameter: SetProperty ( "fat_title_id" ; $fat_title_id ) &
						SetProperty ( "title_environment" ; $title_environment ) &
						SetProperty ( "funkis_cloud_title_id" ; $funkis_cloud_title_id ) ]
