# Called from a button in Title Settings - only developers
# This script export content from FAT and instructs Task Server to build new code versions for all Packages included in Title.
Show Custom Dialog [ Title: "Update content and codebase for all packages" ; Message: "This script will build new versions of both content and code for all packages included in the title. Current code settings will be used.¶¶This will take some time. Make sure that the task server has built all packages by checking your mailbox for the confirmation email for each package. " ; Default Button: "Start build" ; Button 2: "Cancel" ]

If [ Get(lastMessageChoice) = 1 ]
	
	Set Error Capture [ ON ]
	Freeze Window
	Set Variable [ $title_id[1] ; FAT Title::PrimaryKey ]
	Set Variable [ $$success[1] ; 0 ]
	
	Set Variable [ $$error[1] ; 0 ]
	Go to Layout [ Layout: "TABLE FAT Package Alt" ; Animation: None ]
	
	# Exclude Title information
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	
	Go to Record/Request/Page [ First ]
	
	If [ Get(foundCount)>0 ]
		Set Field [ FAT Title::gExportingPackageNum ; 0 ]
		Commit Records/Requests [ With dialog: OFF ]
		// Install OnTimer Script [ "Publish All Remote LOOP CL" ; 30 ]
		// # New (2026-03-20) Task server route can queue the requests, so we don't have to wait, but let's keep a small delay anyway. /Erik
		Install OnTimer Script [ "Publish All Remote LOOP CL" ; 2 ]
		# Start building the first one
		Perform Script [ From list ; "Publish All Remote LOOP CL" ]
	End If
	
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	
End If
