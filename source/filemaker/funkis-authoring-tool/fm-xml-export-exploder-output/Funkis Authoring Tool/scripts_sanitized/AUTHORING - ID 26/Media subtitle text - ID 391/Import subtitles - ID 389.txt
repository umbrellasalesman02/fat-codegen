# Called from media with subtitles layout. 
# Can be called from either card window or standard window
# Active table = FAT Media

# import VTT and SRT only available in FAT Media Subtitle view

Set Error Capture [ ON ]
Freeze Window
# import_mode can be "new" or "update"
Set Variable [ $title_id[1] ; FAT Media::Title id ]
Set Variable [ $package_id[1] ; FAT Media::Package id ]
Set Variable [ $language_code[1] ; Case ( 
				
				    FAT Title::Original localisation code ≠ $$localisation_code; 
				
				        getLanguageCode ( $$localisation_code ) ; 
				
				        Case( FAT Title::Original localisation code ≠ FAT Media::Localisation code;
				
				            FAT Media::Language code;
				
				            FAT Title::Original language code
				
				        )
				
				) ]
Set Variable [ $country_code[1] ; Case ( 
				
				    FAT Title::Original localisation code ≠ $$localisation_code; 
				
				        getCountryCode ( $$localisation_code ) ; 
				
				        Case( FAT Title::Original localisation code ≠ FAT Media::Localisation code;
				
				            FAT Media::Country code;
				
				            FAT Title::Original country code
				
				        )
				
				) ]
Set Variable [ $localisation_code[1] ; $language_code & "-" & $country_code ]
Set Variable [ $media_id[1] ; FAT Media::PrimaryKey ]

# Target must be declared
Commit Records/Requests [ With dialog: OFF ]

If [ FAT Title::Original localisation code ≠ $$localisation_code ]
	Show Custom Dialog [ Title: "Localised subtitle import" ; Message: "You are importing new subtitles to the localised version " & $$localisation_code & ".¶¶"   &
					"Make sure your selected subtitle file is in this language version." & "¶¶"   &
					"Any existing subtitles will be removed." ; Default Button: "OK" ; Button 2: "Cancel" ]
	Set Variable [ $localisation[1] ; True ]
Else If [ FAT Title::Original localisation code ≠ FAT Media::Localisation code ]
	Show Custom Dialog [ Title: "Localised subtitle import" ; Message: "You are importing new subtitles to the localised media with code " & $$localisation_code & ".¶¶"   &
					"Make sure your selected subtitle file is in this language version." & "¶¶"   &
					"Any existing subtitles will be removed." ; Default Button: "OK" ; Button 2: "Cancel" ]
	Set Variable [ $localisation[1] ; True ]
End If


# ----------------- VTT or SRT file -------------------------------------
If [ Platform = "WebDirect" ]
	Go to Field [ Select/perform ; FAT Media::gTempFile ]
	Insert File [ ⚠️ PARAMETER "Options" NOT PARSED ⚠️ ; Target: FAT Media::gTempFile ]
	
	If [ Get(lastError) = 1 ]
		# User cancelled
		Exit Script []
	End If
	
	Set Variable [ $importedSUB[1] ; TextDecode ( FAT Media::gTempFile ; "utf-8" ) ]
	Set Variable [ $filename[1] ; FAT Media::gTempFile ]
	Set Variable [ $format[1] ; Right ( $filename ; 3 ) ]
	
Else
	Set Variable [ $fileInfo[1] ; "" ]
	# Open file dialog box
	Insert File [ ⚠️ PARAMETER "Options" NOT PARSED ⚠️ ; Target: $fileInfo ]
	
	# Example $fileInfo result:
	  file:3_3_quiz_test.en-global.vtt
	  filemac:/Macintosh HD/Users/jensthorsen/Desktop/CMT Tester/TMH filmer & subtitles/content/video/3_3_quiz_test.en-global.vtt
	
	If [ Get(lastError) = 1 ]
		# User cancelled
		Exit Script []
	End If
	Set Variable [ $filepath[1] ; GetValue ( $fileInfo ; 2 ) ]
	Set Variable [ $filename[1] ; GetFileNameFromPath ( $filepath ) ]
	Open Data File [ ⚠️ PARAMETER "UniversalPathList" NOT PARSED ⚠️ ; Target: $fileID ]
	Read from Data File [ ⚠️ PARAMETER "id" NOT PARSED ⚠️ ; Target: $importedSUB ]
	Close Data File [ ⚠️ PARAMETER "id" NOT PARSED ⚠️ ]
	# $format = "VTT" or "SRT"
	Set Variable [ $format[1] ; Right ( $filepath ; 3 ) ]
	Set Variable [ $relative_filepath[1] ; Left ( $filepath ; Position ( $filepath ; $filename ; 0 ; 1 )-1) ]
End If

Go to Layout [ Layout: "TABLE FAT Text" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
If [ Get(foundCount)>0 And $localisation ]
	Delete All Records [ With dialog: OFF ]
End If

If [ ($format = "VTT" And PatternCount ( GetValue ( $importedSUB ; 1 ) ; "WEBVTT" )) Or ($format = "SRT" And GetAsNumber(GetValue ( $importedSUB ; 1 )) =1 ) ]
	Set Variable [ $i[1] ; 1 ]
	Set Variable [ $numOfMediaSubtitles[1] ; 0 ]
	Loop [ Flush: Always ]
		Loop [ Flush: Always ]
			Set Variable [ $i[1] ; $i +1 ]
			Set Variable [ $textLine[1] ; GetValue ( $importedSUB ; $i ) ]
			Exit Loop If [ PatternCount($textLine ; "-->") ]
		End Loop
		If [ $format = "VTT" ]
			Set Variable [ $enter[1] ; GetAsTime (Substitute ( LeftWords ( $textLine ; 2 ) ; "." ; "," )) ]
			Set Variable [ $exit[1] ; GetAsTime (Substitute ( MiddleWords ( $textLine ; 3 ; 2 ) ; "." ; "," ) ) ]
		Else If [ $format = "SRT" ]
			Set Variable [ $enter[1] ; GetAsTime (LeftWords ( $textLine ; 1 )) ]
			Set Variable [ $exit[1] ; GetAsTime ( MiddleWords ( $textLine ; 2 ; 3 )) ]
		End If
		Set Variable [ $i[1] ; $i +1 ]
		Set Variable [ $text[1] ; GetValue ( $importedSUB ; $i ) ]
		Set Variable [ $i[1] ; $i +1 ]
		# Text in VTT can be on two lines
		If [ GetValue ( $importedSUB ; $i ) ≠ "" ]
			Set Variable [ $text[1] ; $text & ¶  &  GetValue ( $importedSUB ; $i ) ]
			Set Variable [ $i[1] ; $i+1 ]
		End If
		
		# Update import log
		Set Variable [ $numOfMediaSubtitles[1] ; $numOfMediaSubtitles + 1 ]
		
		New Record/Request
		Set Field [ FAT Text::Title id ; $title_id ]
		Set Field [ FAT Text::Package id ; $package_id ]
		Set Field [ FAT Text::Text ; $text ]
		Set Field [ FAT Text::Enter time ; $enter ]
		Set Field [ FAT Text::Exit time ; $exit ]
		Set Field [ FAT Text::Parent id ; $media_id ]
		Set Field [ FAT Text::Source item type ; "media" ]
		Set Field [ FAT Text::Source reference ; "Subtitle VTT" ]
		Set Field [ FAT Text::Language code ; $language_code ]
		Set Field [ FAT Text::Country code ; $country_code ]
		Set Field [ FAT Text::Status ; "Approved" ]
		
		Exit Loop If [ $i ≥ ValueCount ( $importedSUB ) ]
	End Loop
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	// Show Custom Dialog [ Title: "Subtitle import result" ; Message: $numOfMediaSubtitles & " subtitles texts were successfully imported" ; Default Button: "OK" ]
Else
	Show Custom Dialog [ Title: "Cannot parse content in file" ; Message: "Sorry but the content in the selected file does not seem to be a valid VTT or SRT file." ; Default Button: "OK" ]
	Go to Layout [ Layout: <original layout> ; Animation: None ]
End If
