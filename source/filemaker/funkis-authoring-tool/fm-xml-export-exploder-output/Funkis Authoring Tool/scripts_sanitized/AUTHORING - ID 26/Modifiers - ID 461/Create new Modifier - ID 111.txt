# This script is called from 3 locations:
# 1. Title settings - Button Create new modifier - level = package, chapter, page or page-item
# 2. Block settings - Button Create new modifier - level = block
# 3. New Modifier Template - looping through all Titles using the template and creating new modifier for each
# From all locations, the script can refer to FAT Title global fields

Freeze Window
Set Error Capture [ ON ]
Commit Records/Requests [ With dialog: OFF ]

Set Variable [ $learningprogramme_id[1] ; GetScriptProperty ( "learningprogramme_id" ) ]
Set Variable [ $level[1] ; GetScriptProperty ( "level" ) ]
Set Variable [ $reset_global_fields[1] ; GetScriptProperty ( "reset_global_fields" ) ]
Set Variable [ $attributeTemplate_id[1] ; GetScriptProperty ( "attributeTemplate_id" ) ]

Set Variable [ $attribute_name[1] ; PrettifyUnderScore (FAT Title::gModifierName) ]
Set Variable [ $attribute_type[1] ; FAT Title::gModifierType ]
Set Variable [ $modifier_values[1] ; FAT Title::gModifierValues ]
Set Variable [ $attribute_default_value[1] ; FAT Title::gModifierDefaultValue ]
Set Variable [ $attribute_description[1] ; FAT Title::gModifierDescription ]
Set Variable [ $modifier_package_type[1] ; FAT Title::gPackageType ]

If [ $level = "Block" ]
	Set Variable [ $symbol_id[1] ; FAT Block::PrimaryKey ]
End If

# Check if attribute name is a duplicate
Perform Script [ From list ; "Check duplicate name" ; Parameter: SetProperty ( "learningprogramme_id" ; $learningprogramme_id ) &
				SetProperty ( "level" ; $level ) &
				SetProperty ( "symbol_id" ; $symbol_id ) &
				SetProperty ( "modifier_package_type" ; $modifier_package_type ) &
				SetProperty ( "name" ; $attribute_name ) ]
If [ Get(ScriptResult)=True ]
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	Show Custom Dialog [ Title: "Modifier name already exist" ; Message: "Please choose another name for your modifier. " ; Default Button: "OK" ]
	Exit Script []
End If

Go to Layout [ Layout: "TABLE FAT Modifier" ; Animation: None ]
# Create the attribute
New Record/Request
Set Variable [ $attribute_id[1] ; FAT Modifier::PrimaryKey ]
Set Field [ FAT Modifier::Title id ; $learningprogramme_id ]
Set Field [ FAT Modifier::Parent id ; $symbol_id ]
Set Field [ FAT Modifier::Name ; $attribute_name ]
Set Field [ FAT Modifier::Type ; $attribute_type ]
Set Field [ FAT Modifier::Level ; $level ]
Set Field [ FAT Modifier::Values ; $modifier_values ]
Set Field [ FAT Modifier::DefaultValue ; $attribute_default_value ]
Set Field [ FAT Modifier::Description ; $attribute_description ]
Set Field [ FAT Modifier::ModifierTemplate id ; $attributeTemplate_id ]
Set Field [ FAT Modifier::Package type id ; $modifier_package_type ]
Commit Records/Requests [ With dialog: OFF ]

# Add attribute to all existing items if not Title. For Title it will be ValueLists that will be added manually under tab IO
If [ $level ≠ "Title" ]
	Perform Script [ From list ; "Add modifier to existing items" ; Parameter: SetProperty ( "lp_id" ; $learningprogramme_id ) &
					SetProperty ( "level" ; $level ) &
					SetProperty ( "modifier_package_type" ; $modifier_package_type )  &
					SetProperty ( "symbol_id" ; $symbol_id )  &
					SetProperty ( "attribute_type" ; $attribute_type )  &
					SetProperty ( "modifier_id" ; $attribute_id )  &
					SetProperty ( "modifier_name" ; $attribute_name)  &
					SetProperty ( "modifier_default_value" ; $attribute_default_value ) ]
End If

If [ $reset_global_fields = True ]
	Set Field [ FAT Title::gModifierName ; "" ]
	Set Field [ FAT Title::gModifierType ; "" ]
	Set Field [ FAT Title::gModifierLevel ; "" ]
	Set Field [ FAT Title::gModifierValues ; "" ]
	Set Field [ FAT Title::gModifierDefaultValue ; "" ]
	Set Field [ FAT Title::gModifierDescription ; "" ]
	Set Field [ FAT Title::gPackageType ; "" ]
End If
Go to Layout [ Layout: <original layout> ; Animation: None ]
Go to Object [ "Modifier" ; Repetition: 1 ]
