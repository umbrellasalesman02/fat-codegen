# For modifiers using a value list, this script deletes the valu
Set Variable [ $mode[1] ; Get(scriptParameter) ]
If [ $mode = "Title" ]
	Set Field [ FAT ModifierValue @ Title::Value ; "" ]
Else If [ $mode = "Package" ]
	Set Field [ FAT ModifierValue @ Package::Value ; "" ]
Else If [ $mode = "Chapter" ]
	Set Field [ FAT ModifierValue @ Chapter::Value ; "" ]
Else If [ $mode = "Page" ]
	Set Field [ FAT ModifierValue @ Page::Value ; "" ]
Else If [ $mode = "Page item" ]
	Set Field [ FAT ModifierValue @ Page item::Value ; "" ]
Else If [ $mode = "IO" ]
	Set Field [ FAT ModifierValue @ IO::Value ; "" ]
End If
