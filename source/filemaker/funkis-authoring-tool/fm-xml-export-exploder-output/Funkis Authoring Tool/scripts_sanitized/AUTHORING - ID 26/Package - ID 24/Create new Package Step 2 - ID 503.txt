# Called from a button in card window Create Package
# Also called from script "Create new Title Step 2 SRV" - when using Funkis Cloud a Title Site is automatically created
Set Variable [ $package_id[1] ; Get(scriptParameter) ]

Freeze Window
If [ Not IsEmpty ( $package_id ) ]
	Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
End If

If [ IsEmpty ( FAT Package::Reference name ) ]
	Show Custom Dialog [ Title: "Reference name needed" ; Message: "You need to have a reference name for your package. The reference name must be unique in the title. " ; Default Button: "OK" ]
	Exit Script []
End If

Set Variable [ $title_id[1] ; FAT Title::PrimaryKey ]
Set Variable [ $package_id[1] ; FAT Package::PrimaryKey ]
Set Variable [ $reference_title[1] ; FAT Package::Reference name ]
Set Field [ FAT Package::FC Package directory ; TextFormatRemove(PrettifyDash($reference_title)) ]

# Cannot add overview until user selected package type
# Add to content overview - NEW 2024-02-10
Set Variable [ $item_type[1] ; "Package" ]
Set Variable [ $item_id[1] ; FAT Package::PrimaryKey ]
Perform Script [ From list ; "Add Overview item SRV" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
				SetProperty ( "item_id" ; $item_id ) &
				SetProperty ( "item_type" ; $item_type ) ]

If [ PatternCount (Get(windowName); "new")
				// New Package ]
	Close Window
End If

Perform Script [ From list ; "Save Active Layout" ]
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
Set Variable [ $package_type[1] ; FAT Package::Package type id ]
Set Variable [ $title_id[1] ; FAT Package::Parent id ]


# All blocks, attributes and stateToggles should already be imported since they belong to the learning programme
# Only add modifiers that corresponds with the package type
Perform Script [ From list ; "Add Modifiers to item SRV" ; Parameter: SetProperty ( "level" ; "Package" ) & 
				SetProperty ( "item_id" ; $package_id ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $package_id ) &
				SetProperty ( "package_type" ; $package_type ) ]
Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Package::PrimaryKey ) &
				SetProperty ( "source_table" ; "package" ) &
				SetProperty ( "source_textfield" ; GetFieldName ( FAT Package::Header txt id ) ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $package_id ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]
Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Package::PrimaryKey ) &
				SetProperty ( "source_table" ; "package" ) &
				SetProperty ( "source_textfield" ; GetFieldName ( FAT Package::Sub header 1 txt id ) ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $package_id ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]
Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Package::PrimaryKey ) &
				SetProperty ( "source_table" ; "package" ) &
				SetProperty ( "source_textfield" ; GetFieldName ( FAT Package::Sub header 2 txt id ) ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "package_id" ; $package_id ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]

# Depending on package type
  
  PrimaryKey							Name			JSON name
  4C393C7E-B339-4BFA-ADE7-E59F47FBAC60	Scroll		scroll
  5F127BA8-1BC7-467B-93EF-7BE401615BAA	Slide		slide
  4B8AB81A-9C99-43AD-8C32-3325D98F688A	Mixed type	mixed type
  BB6DDA20-C6CE-441E-B99C-CE1F884DC7E5	Site	 old		site
  B6397ED6-1BA8-4444-BD51-65076EE684F9	Site			scroll
  C69777FA-7286-422C-9337-B722B10663B6  Title Info   title-information

Set Variable [ $title_codebase[1] ; Case ( Not IsEmpty (FAT Title::Codebase id) ; FAT Title::Codebase id; "547B3496-8B9F-45C5-90B1-5C712DD063EB" /* Readable name: Funkis Player @ Latest */) ]
Set Variable [ $title_codebase_version[1] ; FAT Title::Codebase version id ]

If [ $package_type = "BB6DDA20-C6CE-441E-B99C-CE1F884DC7E5" Or $package_type = "B6397ED6-1BA8-4444-BD51-65076EE684F9" ]
	If [ $package_type = "BB6DDA20-C6CE-441E-B99C-CE1F884DC7E5" ]
		# Site old
		Set Field [ FAT Package::Codebase id ; "9A8219CF-FC37-4E87-BDFE-836264287CB5" /* Base linked - Readable name = Title Site + Latest unstable */ ]
	Else
		# Site
		Set Field [ FAT Package::Codebase id ; $title_codebase ]
		Set Field [ FAT Package::Codebase version id ; $title_codebase_version ]
	End If
	
	# Force Funkis Cloud if trying to build a Site - this will be removed when adding a Title directly from FAT
	Set Field [ FAT Title::Title Backend ; "Funkis Cloud" ]
	
Else If [ $package_type = "4C393C7E-B339-4BFA-ADE7-E59F47FBAC60" Or $package_type = "5F127BA8-1BC7-467B-93EF-7BE401615BAA" ]
	# Scroll or Slide
	Set Field [ FAT Package::Codebase id ; $title_codebase ]
	Set Field [ FAT Package::Codebase version id ; $title_codebase_version ]
Else If [ $package_type = "C69777FA-7286-422C-9337-B722B10663B6" // Title info ]
	# Title info
	Set Variable [ $title_info_package_type[1] ; "C69777FA-7286-422C-9337-B722B10663B6" ]
	
	# Check if it already exist a Title info package in Title
	Go to Layout [ Layout: "TABLE FAT Package Alt" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	If [ Get(foundCount)>0 ]
		# A Title info package already exist
		Set Variable [ $title_info_package_id[1] ; FAT Package Alt::PrimaryKey ]
		Go to Layout [ Layout: "FAT Package Settings" ; Animation: None ]
		Show Custom Dialog [ Title: "Cannot create Title info" ; Message: "You can only have one Title info package in your Title." ; Default Button: "OK" ]
		Freeze Window
		# Delete the new package and go to existing
		Delete Record/Request [ With dialog: OFF ]
		Perform Find [ ⚠️ RESTORE ⚠️ ]
	Else
		Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
	End If
	# Set reference name
	Set Field [ FAT Package::Reference name ; "Title information" // Name is not allowed to change for this package type ]
	Set Variable [ $reference_title[1] ; "Title information" // Name is not allowed to change ]
Else
	# Mixed type
End If
Set Field [ TF PK Header::Text ; $reference_title ]

If [ $package_type ≠ "4B8AB81A-9C99-43AD-8C32-3325D98F688A" And $package_type ≠ "C69777FA-7286-422C-9337-B722B10663B6" ]
	# If neither a Title information nor a Mixed type package
	# Instruct Task Server to perform the first build
	Perform Script [ From list ; "Publish Remote CL" ; Parameter: SetProperty ( "publish_mode" ; "CODE_UPDATE" ) &
					SetProperty ( "caller" ; "Package") &
					SetProperty ( "show_dialog" ; "false") &
					SetProperty ( "first_build" ; "true") ]
	
	# Create a link to this item
	Perform Script [ From list ; "Find or Create Link" ; Parameter: SetProperty ( "title_id" ; FAT Title::PrimaryKey ) &
					SetProperty ( "item_type" ; "Package" ) &
					SetProperty ( "item_id" ; FAT Package::PrimaryKey ) &
					SetProperty ( "link_type" ; "Internal" ) &
					SetProperty ( "link_name" ; FAT Package::Reference name & " - Link" )&
					SetProperty ( "caller" ; "New package" ) ]
	
End If


Go to Layout [ Layout: "FAT Package Settings" ; Animation: None ]
