# Called from IO card when creating a field of a list type.
# Can also be used to create an aspect list for AI aspects
# List type can be either Choice for dropdown field or Aspect list for Insight.

Set Variable [ $title_id[1] ; GetScriptProperty("title_id") ]
Set Variable [ $field_id[1] ; GetScriptProperty("field_id") ]
Set Variable [ $list_name[1] ; GetScriptProperty("list_name") ]
Set Variable [ $list_type[1] ; GetScriptProperty("list_type") ]
Set Variable [ $list_template_id[1] ; GetScriptProperty("list_template_id") ]

Freeze Window
Set Error Capture [ ON ]

Go to Layout [ Layout: "TABLE FAT List" ; Animation: None ]
New Record/Request
// Set Variable [ $list_template_id[1] ; "BAF6DADF-7209-4916-BD32-980F27F248EA" // Multiple choice ]

Set Variable [ $list_id[1] ; FAT List::PrimaryKey ]
Set Field [ FAT List::Title id ; $title_id ]
Set Field [ FAT List::List template id ; $list_template_id ]
Set Field [ FAT List::List backend ; "FAT" ]

Commit Records/Requests [ With dialog: OFF ]

If [ List template::List has modifiers ]
	Perform Script [ From list ; "Add Modifiers to item SRV" ; Parameter: SetProperty ( "level" ; "List" ) & 
					SetProperty ( "item_id" ; $list_id ) &
					SetProperty ( "learningprogramme_id" ; $title_id ) &
					SetProperty ( "symbol_id" ; $list_template_id ) ]
End If

Perform Script [ From list ; "Set valid object name" ; Parameter: SetProperty ( "source_textfield" ; GetFieldName ( FAT List::Name ) ) &
				SetProperty ( "new_name" ; $list_name ) &
				SetProperty ( "max_characters" ; 32 ) ]


Go to Layout [ Layout: "TABLE FAT IO Field" ; Animation: None ]
Perform Find [ ⚠️ RESTORE ⚠️ ]
If [ $list_type = "aspect" ]
	Set Field [ FAT IO Field::Aspect List id ; $list_id ]
Else
	Set Field [ FAT IO Field::List id ; $list_id ]
End If

Go to Layout [ Layout: <original layout> ; Animation: None ]
Perform Script [ From list ; "View Card List Setting" ; Parameter: SetProperty ("list_id"; $list_id  ) ]
