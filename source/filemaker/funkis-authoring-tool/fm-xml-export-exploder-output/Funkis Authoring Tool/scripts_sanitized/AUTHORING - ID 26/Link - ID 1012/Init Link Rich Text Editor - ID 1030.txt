# Link type can be:
  Internal in Title
  External URL
  File download

Set Variable [ $Link_HTML_JSON[1] ; /* HTML Parameters */ 
				
				
				Let ( 
				       [
				         i = 1;
				         trans = Case ( FAT Title::gViewTranslatedLanguage ; " TRANS" ; "" );
				         localisation_mode = Case ( FAT Title::gViewTranslatedLanguage; "translate"; "original");
				         localisation_code = $$localisation_code;
				      
				
				        LKjson2 = JSONSetElement ( "{}"  ;
				             ["label" ; "Link header" ; JSONString];
				             ["id" ; "2" ; JSONString];
				             ["html" ; True ; JSONBoolean];
				             ["maxchars" ; 10000 ; JSONNumber];
				             ["value" ; sanitizeHTML ( GetField("TF LK 1" & trans & "::Text") ) ; JSONString] 
				         );
				
				        LKjson3 = JSONSetElement ( "{}"  ;
				             ["label" ; "Link body" ; JSONString];
				             ["id" ; "3" ; JSONString];
				             ["html" ; True ; JSONBoolean];
				             ["maxchars" ; 10000 ; JSONNumber];
				             ["value" ; sanitizeHTML ( GetField("TF LK 2" & trans & "::Text") ) ; JSONString] 
				         );
				
				         LKtextArr = "[" & LKjson2 & "," & LKjson3  & "]";
				 
				          
				         LKjsonResult = JSONSetElement ( "{}" ; 
				
				                      ["environment" ; Case ( Platform="WebDirect"; "web"; "FileMaker") ; JSONString];
				                      ["record-id" ; Get(recordID) ; JSONNumber];
				                      ["localisation_mode" ; localisation_mode; JSONString];
				                      ["edit_mode" ; "titleLink"; JSONString];
				                      ["localisation_code" ; localisation_code ; JSONString];
				                      ["texts" ; LKtextArr ; JSONArray]
				               )
				
				       
				
				        ];
				
				      JSONFormatElements ( LKjsonResult)
				) ]
