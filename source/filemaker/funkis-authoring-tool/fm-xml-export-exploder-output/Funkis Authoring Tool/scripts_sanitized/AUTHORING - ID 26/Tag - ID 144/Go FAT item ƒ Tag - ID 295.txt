Perform Script [ From list ; "Save Active Layout" ]
Set Error Capture [ ON ]
Freeze Window
Set Variable [ $item_id[1] ; FAT Tag Value::Parent id ]
Set Variable [ $level[1] ; FAT Tag Value::Level ]
If [ $level = "Package" ]
	Go to Layout [ Layout: "FAT Package" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
Else If [ $level = "Chapter" ]
	Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $course_id[1] ; FAT Chapter::Parent id ]
	Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Go to Layout [ Layout: "FAT Chapter" ; Animation: None ]
Else If [ $level = "Page" ]
	Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $chapter_id[1] ; FAT Page::Parent id ]
	Go to Layout [ Layout: "TABLE FAT Chapter" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Set Variable [ $course_id[1] ; FAT Chapter::Parent id ]
	Go to Layout [ Layout: "TABLE FAT Package" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Go to Layout [ Layout: "FAT Page" ; Animation: None ]
Else If [ $level = "Page item" ]
	Set Variable [ $$pageItemRecordLoadTrigger[1] ; True ]
	Go to Layout [ Layout: "TABLE FAT Page Item" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Go to Layout [ Layout: "FAT Page item" ; Animation: None ]
Else If [ $level = "List item" ]
	Go to Layout [ Layout: "FAT TT List Item Card" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
End If
