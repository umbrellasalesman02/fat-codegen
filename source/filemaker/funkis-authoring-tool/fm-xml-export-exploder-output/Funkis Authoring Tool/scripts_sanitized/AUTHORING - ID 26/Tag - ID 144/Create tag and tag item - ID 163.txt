# Called from any layout CMT Package, Chapter, Page, Page item
# Create both a tag and tags the item

Set Variable [ $level[1] ; Get(scriptParameter) ]
Set Variable [ $labeltext[1] ; FAT Title::gTagName ]
Set Variable [ $title_id[1] ; FAT Title::PrimaryKey ]
If [ $level= "Title" ]
	Set Variable [ $item_id[1] ; FAT Title::PrimaryKey ]
Else If [ $level = "Package" ]
	Set Variable [ $item_id[1] ; FAT Package::PrimaryKey ]
Else If [ $level = "Chapter" ]
	Set Variable [ $item_id[1] ; FAT Chapter::PrimaryKey ]
Else If [ $level = "Page" ]
	Set Variable [ $item_id[1] ; FAT Page::PrimaryKey ]
Else If [ $level = "Page item" ]
	Set Variable [ $item_id[1] ; FAT Page item::PrimaryKey ]
End If
Go to Layout [ Layout: "TABLE FAT Tag" ; Animation: None ]
New Record/Request
Set Field [ FAT Tag::Title id ; $title_id ]
Set Field [ FAT Tag::Level ; $level ]
Set Variable [ $label_id[1] ; FAT Tag::PrimaryKey ]
Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Tag::PrimaryKey ) &
				SetProperty ( "source_table" ; "tag" ) &
				SetProperty ( "learningprogramme_id" ; $title_id ) &
				SetProperty ( "source_textfield" ; GetFieldName (FAT Tag::Name text id) ) &
				SetProperty ( "caller_layout" ; Get(layoutName )) ]
Set Field [ TF TAG Name::Text ; $labeltext ]
Go to Layout [ Layout: "TABLE FAT Tag Value" ; Animation: None ]
New Record/Request
Set Field [ FAT Tag Value::Tag id ; $label_id ]
Set Field [ FAT Tag Value::Level ; $level ]
Set Field [ FAT Tag Value::Parent id ; $item_id ]
Go to Layout [ Layout: <original layout> ; Animation: None ]
Set Field [ FAT Title::gTagName ; "" ]
Go to Object [ "tag_tab" ; Repetition: 1 ]
