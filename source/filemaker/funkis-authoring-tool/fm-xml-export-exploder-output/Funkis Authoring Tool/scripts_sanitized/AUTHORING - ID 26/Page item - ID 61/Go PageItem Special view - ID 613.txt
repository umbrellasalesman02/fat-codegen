Freeze Window
Set Error Capture [ ON ]
If [ PatternCount ( FAT Block @ Page item::Name ; "Quiz") ]
	# Go to assigned quiz page
	Set Variable [ $page_id[1] ; Case ( 
					
					   // It exist two quiz attributes depending on block. 
					
					
					   JSONGetElement ( FAT Page item::Page item JSON ; "quiz_page" ) = "";
					
					   JSONGetElement ( FAT Page item::Page item JSON ; "page_id" ) ;
					
					   JSONGetElement ( FAT Page item::Page item JSON ; "quiz_page" )
					
					) ]
	Set Variable [ $$pageItemRecordLoadTrigger[1] ; False ]
	Perform Script [ From list ; "Open card setting" ; Parameter: SetProperty ("card_window_name" ; "Quiz questions" ) &
					SetProperty ("card_window_layout" ; "FAT Page item Card" ) &
					SetProperty ("card_window_width" ; "1140" ) &
					SetProperty ("card_window_height" ; "680" ) &
					SetProperty ("item_id" ; $page_id ) &
					SetProperty ("item_type" ; "Sub page item" ) ]
	If [ Get(foundCount) > 0 ]
		Set Variable [ $$pageItemRecordLoadTrigger[1] ; True ]
		Perform Script [ From list ; "Show page item 2" ]
	End If
	
Else If [ PatternCount ( FAT Block @ Page item::Name ; "Site Activity") ]
	Perform Script [ From list ; "Save Active Layout" ]
	# Go to the package that the site link refers to - MUST be updated
	Set Variable [ $package_id[1] ; JSONGetElement ( FAT Page item::Page item JSON ; "package_id" ) ]
	Go to Layout [ Layout: "TABLE FAT Title" ; Animation: None ]
	Go to Related Record [ Table: FAT Package ; Layout: "TABLE FAT Package" ; Show related ]
	Loop [ Flush: Always ]
		Exit Loop If [ FAT Package::PrimaryKey = $package_id ]
		Go to Record/Request/Page [ Next ; Exit after last: ON ]
	End Loop
	If [ FAT Package::PrimaryKey = $package_id ]
		Go to Layout [ Layout: "FAT Package" ; Animation: None ]
	Else
		Go to Layout [ Layout: <original layout> ; Animation: None ]
	End If
Else If [ PatternCount ( FAT Block @ Page item::Name ; "Activity link") ]
	Perform Script [ From list ; "Save Active Layout" ]
	# Go to the package that the site link refers to - MUST be updated
	Set Variable [ $link_id[1] ; JSONGetElement ( FAT Page item::Page item JSON ; "activity_link_fat_link_id" ) ]
	Go to Layout [ Layout: "TABLE FAT Link" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	If [ Get(foundCount) = 1 ]
		
		Set Variable [ $level[1] ; FAT Link::Internal item type ]
		Set Variable [ $title_id[1] ; FAT Link::Title id ]
		Set Variable [ $item_id[1] ; FAT Link::Internal Item id ]
		Perform Script [ From list ; "Go FAT Item" ; Parameter: SetProperty ( "level" ; $level ) &
						SetProperty ( "title_id" ; $title_id ) &
						SetProperty ( "item_id" ; $item_id ) ]
	Else
		Go to Layout [ Layout: <original layout> ; Animation: None ]
		Show Custom Dialog [ Title: "Cannot find target" ; Message: "This activity link does not seem to be valid. Maybe it is not set." ; Default Button: "OK" ]
	End If
	
	
	
Else
	Perform Script [ From list ; "Save Active Layout" ]
	Go to Layout [ Layout: "FAT Page item Special" ; Animation: None ]
End If
