Freeze Window

# Called from Create new List and more
Set Variable [ $list_id[1] ; GetScriptProperty ( "list_id" ) ]
Set Variable [ $title_id[1] ; FAT Title::PrimaryKey ]
Commit Records/Requests [ With dialog: OFF ]

Perform Script [ From list ; "Open card setting" ; Parameter: SetProperty ("card_window_name" ; "List settings" ) &
				SetProperty ("card_window_layout" ; "FAT TT List Card" ) &
				SetProperty ("card_window_width" ; "1140" ) &
				SetProperty ("card_window_height" ; "680" ) &
				SetProperty ("item_id" ; $list_id ) &
				SetProperty ("item_type" ; "List" ) ]

# Set the the right Tag list for list items (should be page item.)
Set Field [ FAT Title::gLevel ; "Page item" ]
# This is set in order to for IO popup list to only show IOs of type = Post.
Set Field [ FAT Title::gIOFilter ; "Post" ]

Freeze Window
Perform Find [ ⚠️ RESTORE ⚠️ ]
# If List is a Group of list - view in another layout
If [ FAT List::List of Lists ]
	Go to Layout [ Layout: "FAT TT List Group Card" ; Animation: None ]
End If
