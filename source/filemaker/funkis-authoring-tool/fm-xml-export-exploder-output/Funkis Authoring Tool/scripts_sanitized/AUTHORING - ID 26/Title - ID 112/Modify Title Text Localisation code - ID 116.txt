Show Custom Dialog [ Title: "Modify localisation code on all texts" ; Message: "By changing the language and / or country code, all text fields in the whole program needs to be updated. ¶¶Is this OK?" ; Default Button: "OK" ; Button 2: "Cancel" ]
If [ Get(lastMessageChoice) = 1 ]
	Freeze Window
	Set Error Capture [ ON ]
	Set Variable [ $lp_id[1] ; FAT Title::PrimaryKey ]
	Set Variable [ $language[1] ; FAT Title::Original language code ]
	Set Variable [ $country[1] ; FAT Title::Original country code ]
	Set Variable [ $localisation_code[1] ; FAT Title::Original localisation code ]
	
	Go to Layout [ Layout: "TABLE User X Title" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Replace Field Contents [ User X Title::Language ; Replace with calculation: $language ]
	Replace Field Contents [ User X Title::Localisation code ; Replace with calculation: $localisation_code ]
	
	Go to Layout [ Layout: "TABLE FAT Text" ; Animation: None ]
	Perform Find [ ⚠️ RESTORE ⚠️ ]
	Replace Field Contents [ FAT Text::Language code ; Replace with calculation: $language ]
	Replace Field Contents [ FAT Text::Country code ; Replace with calculation: $country ]
	
	// Go to Layout [ Layout: "TABLE FAT Media" ; Animation: None ]
	// Perform Find [ ⚠️ RESTORE ⚠️ ]
	// Replace Field Contents [ FAT Text::Language code ; Replace with calculation: $language ]
	// Replace Field Contents [ FAT Text::Country code ; Replace with calculation: $country ]
	
	Go to Layout [ Layout: <original layout> ; Animation: None ]
	Set Variable [ $$localisation_code[1] ; FAT Title::Original localisation code ]
End If
