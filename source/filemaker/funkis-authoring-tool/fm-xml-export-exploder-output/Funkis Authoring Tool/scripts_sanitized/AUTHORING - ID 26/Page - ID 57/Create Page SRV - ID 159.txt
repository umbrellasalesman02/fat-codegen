# This script is called from two locations:
# 1. From the Chapter layout - Create Page CL
# 2. From the Page item layout - Create PageItem in Page CL
Set Error Capture [ ON ]
Set Variable [ $learningprogramme_id[1] ; GetScriptProperty ( "learningprogramme_id" ) ]
Set Variable [ $title_id[1] ; $learningprogramme_id ]
Set Variable [ $parent_id[1] ; GetScriptProperty ( "parent_id" ) ]
Set Variable [ $proxy_type[1] ; GetScriptProperty ( "proxy_type" ) ]
Set Variable [ $block_id[1] ; GetScriptProperty ( "block_id" ) ]
Set Variable [ $sort_index[1] ; GetScriptProperty ( "sort_index" ) ]
Set Variable [ $hierarchy_level[1] ; GetScriptProperty ( "hierarchy_level" ) ]

If [ $sort_index ≠ "" ]
	Perform Script [ From list ; "GetNextSortIndex" ; Parameter: SetProperty ( "sort_table" ; "page" ) &
					SetProperty ( "parent_id" ; $parent_id ) &
					SetProperty ( "sort_index" ; $sort_index ) ]
	Set Variable [ $new_index[1] ; GetAsNumber (Get(ScriptResult)) ]
End If
Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
New Record/Request
Set Field [ FAT Page::Parent id ; $parent_id ]
Set Variable [ $page_id[1] ; FAT Page::PrimaryKey ]
Set Variable [ $package_type[1] ; FAT Package::Package type id ]
Set Variable [ $package_id[1] ; FAT Package::PrimaryKey ]

If [ $proxy_type = "" ]
	Set Field [ FAT Page::Hierarchy level ; $hierarchy_level + 1 ]
	Perform Script [ From list ; "Add Modifiers to item SRV" ; Parameter: SetProperty ( "level" ; "Page" ) & 
					SetProperty ( "item_id" ; $page_id ) &
					SetProperty ( "learningprogramme_id" ; $learningprogramme_id ) &
					SetProperty ( "package_id" ; $package_id ) &
					SetProperty ( "package_type" ; $package_type ) ]
	Go to Layout [ Layout: "TABLE FAT Page" ; Animation: None ]
	Perform Script [ From list ; "New media" ; Parameter: SetProperty ( "location_id" ; FAT Page::PrimaryKey ) &
					SetProperty ( "source_table" ; "page" ) &
					SetProperty ( "source_mediafield" ; GetFieldName ( FAT Page::Main menu media id ) ) &
					SetProperty ( "caller_layout" ; Get(layoutName )) &
					SetProperty ( "package_id" ; $package_id ) &
					SetProperty ( "learningprogramme_id" ; $learningprogramme_id) ]
	Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Page::PrimaryKey ) &
					SetProperty ( "source_table" ; "page" ) &
					SetProperty ( "source_textfield" ; GetFieldName ( FAT Page::Header txt id ) ) &
					SetProperty ( "caller_layout" ; Get(layoutName )) ]
	Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Page::PrimaryKey ) &
					SetProperty ( "source_table" ; "page" ) &
					SetProperty ( "source_textfield" ; GetFieldName ( FAT Page::Sub header 1 txt id ) ) &
					SetProperty ( "caller_layout" ; Get(layoutName )) ]
	Perform Script [ From list ; "New text" ; Parameter: SetProperty ( "location_id" ; FAT Page::PrimaryKey ) &
					SetProperty ( "source_table" ; "page" ) &
					SetProperty ( "source_textfield" ; GetFieldName ( FAT Page::Sub header 2 txt id ) ) &
					SetProperty ( "caller_layout" ; Get(layoutName )) ]
	
	# Add to content overview - NEW 2024-02-10
	Set Variable [ $item_type[1] ; "Page" ]
	Set Variable [ $item_id[1] ; $page_id ]
	Perform Script [ From list ; "Add Overview item SRV" ; Parameter: SetProperty ( "title_id" ; $title_id ) &
					SetProperty ( "item_id" ; $item_id ) &
					SetProperty ( "item_type" ; $item_type ) ]
	
	Exit Script [ $page_id ]
Else If [ $proxy_type = "page_item" ]
	# Special pages cannot be created as proxy pages - page type = "default"
	Set Field [ FAT Page::Proxy type ; $proxy_type ]
	Set Field [ FAT Page::Page type ; "default" ]
	Set Field [ FAT Page::Allowed Block type ; FAT Page::Package type ]
	If [ $new_index ≠ "" ]
		Set Field [ FAT Page::SortIndex ; $new_index ]
	End If
	# This script is PSOS
	Perform Script [ From list ; "Create PageItem SRV" ; Parameter: SetProperty ( "page_id" ; $page_id ) &
					SetProperty ( "block_id" ; $block_id ) &
					SetProperty ( "hierarchy_level" ; $hierarchy_level ) &
					SetProperty ( "sort_index" ; $sort_index ) &
					SetProperty ( "sort_table" ; "page" ) &
					SetProperty ( "learningprogramme_id" ; $learningprogramme_id ) ]
	Go to Layout [ Layout: "TABLE FAT Page Item" ; Animation: None ]
	Set Field [ FAT Page item::Hierarchy level ; $hierarchy_level + 1 ]
	Set Variable [ $pageitem_id[1] ; FAT Page item::PrimaryKey ]
	Exit Script [ $pageitem_id ]
End If
